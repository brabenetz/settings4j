19  import org.settings4j.connector.PropertyFileConnector;
20  import org.settings4j.connector.SystemPropertyConnector;
21  import org.settings4j.contentresolver.ReadOnlyContentResolverWrapper;

24  /**
25   * a ContentResolver is a Helper for read/write byte[] content for a given key.
26   * 
27   * <pre>
28   * Example configuration in settings4j.xml:
29   * --------------------------------------
30   * &lt;contentResolver name="ClasspathContentResolver" class="org.settings4j.contentresolver.ClasspathContentResolver"&gt;
31   * &lt;/contentResolver&gt;
32   * --------------------------------------
33   * </pre>
34   * 
35   * This is usefull for {@link SystemPropertyConnector} or {@link PropertyFileConnector}.
36   * If you define a ContentResolver in this Connectors, you can rever to a File of the FileSystem or Classpath.
37   * 
38   * <pre>
39   * Example Connector usage in settings4j.xml:
40   * --------------------------------------
41   * &lt;connector name="SystemPropertyConnector" class="org.settings4j.connector.SystemPropertyConnector" &gt;
42   *     &lt;contentResolver-ref ref="ClasspathContentResolver" /&gt;
43   * &lt;/connector&gt;
44   * --------------------------------------
45   * 
46   * Example usage in java-code:
47   * 
48   * --------------------------------------
49   * // alternativ start myapp with -Dxyz=com/mycompany/myapp/xyz-config.xml
50   * System.setProperty("xyz", "com/mycompany/myapp/xyz-config.xml"); //refer to the ClasspathContentResolver
51   * 
52   * // somewhere in myapp:
53   * byte[] xyzConfig = Settings.getContent("xyz"); // get Classpath-URL from the SystemPropertyConnector
54   * --------------------------------------
55   * 
56   * </pre>
57   * 
58   * @author hbrabenetz
59   *
60   */
61  public interface ContentResolver {
62      
63      /**
64       * Reads the Content for the given Key or null if nothing where found.
65       * 
66       * @param key The key
67       * @return The byte[] Content or null if nothing where found.
68       */
69      byte[] getContent(String key);
70      
71      /**
72       * Set a new byte[] Content for the given Key.
73       * If the Implementation can write the Content,
74       * then the Connector must return {@link Constants#SETTING_SUCCESS}<br />
75       * 
76       * If there was a Problem, or The Connector doesn't supports write-access,
77       * then the Connector must return {@link Constants#SETTING_NOT_POSSIBLE}
78       * 
79       * @param key The Key for the new content
80       * @param value The new byte[] Content for the given Key
81       * @return {@link Constants#SETTING_SUCCESS} or {@link Constants#SETTING_NOT_POSSIBLE}
82       */
83      int setContent(String key, byte[] value);
84      
85      /**
86       * Some Implementations of a {@link ContentResolver} are delegating the functionality
87       * to other ContentResolvers.<br />
88       * Examples are: {@link ReadOnlyContentResolverWrapper} or {@link UnionContentResolver}
89       * 
90       * <pre>
91       * --------------------------------------
92       * &lt;contentResolver name="DefaultContentResolver" class="org.settings4j.contentresolver.UnionContentResolver"&gt;
93       *     &lt;contentResolver-ref ref="FSContentResolver" /&gt;
94       *     &lt;contentResolver-ref ref="ClasspathContentResolver" /&gt;
95       * &lt;/contentResolver&gt;
96       * --------------------------------------
97       * </pre>
98       * 
99       * @param contentResolver the original contentResolver to delegate.
100      */
101     void addContentResolver(ContentResolver contentResolver);
102 }
