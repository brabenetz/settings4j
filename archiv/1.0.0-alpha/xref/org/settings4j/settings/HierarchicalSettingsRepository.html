60      public List getCurrentSettingsList() {
61          // The accumlation in list is necessary because not all elements in
62          // ht are Logger objects as there might be some ProvisionNodes
63          // as well.
64          List list = new ArrayList(map.size());
65  
66          Iterator iterator = map.values().iterator();
67          while (iterator.hasNext()) {
68              Object o = iterator.next();
69              if (o instanceof Settings) {
70                  list.add(o);
71              }
72          }
73          return list;
74      }

84      public Settings getSettings(String name, SettingsFactory factory) {
85          // System.out.println("getInstance("+name+") called.");
86          SettingsKey key = new SettingsKey(name);
87          // Synchronize to prevent write conflicts. Read conflicts (in
88          // getChainedLevel method) are possible only if variable
89          // assignments are non-atomic.
90          Settings settings;
91  
92          synchronized (map) {
93              Object o = map.get(key);
94              if (o == null) {
95                  settings = factory.makeNewSettingsInstance(name);
96                  settings.setSettingsRepository(this);
97                  map.put(key, settings);
98                  updateParents(settings);
99                  return settings;
100             } else if (o instanceof Settings) {
101                 return (Settings) o;
102             } else if (o instanceof ProvisionNode) {
103                 // System.out.println("("+name+") ht.get(this) returned ProvisionNode");
104                 settings = factory.makeNewSettingsInstance(name);
105                 settings.setSettingsRepository(this);
106                 map.put(key, settings);
107                 updateChildren((ProvisionNode) o, settings);
108                 updateParents(settings);
109                 return settings;
110             } else {
111                 // It should be impossible to arrive here
112                 return null; // but let's keep the compiler happy.
113             }
114         }
115     }

117     /**
118      * This method loops through all the *potential* parents of 'cat'. There 3 possible cases:
119      * 
120      * 1) No entry for the potential parent of 'cat' exists
121      * 
122      * We create a ProvisionNode for this potential parent and insert 'cat' in that provision node.
123      * 
124      * 2) There entry is of type Logger for the potential parent.
125      * 
126      * The entry is 'cat's nearest existing parent. We update cat's parent field with this entry. We
127      * also break from the loop because updating our parent's parent is our parent's responsibility.
128      * 
129      * 3) There entry is of type ProvisionNode for this potential parent.
130      * 
131      * We add 'cat' to the list of children for this potential parent.
132      */
133     final private void updateParents(Settings settings) {
134         String name = settings.getName();
135         int length = name.length();
136         boolean parentFound = false;
137 
138         // System.out.println("UpdateParents called for " + name);
139 
140         // if name = "w.x.y.z", loop thourgh "w.x.y", "w.x" and "w", but not "w.x.y.z"
141         for (int i = name.lastIndexOf('.', length - 1); i >= 0; i = name.lastIndexOf('.', i - 1)) {
142             String substr = name.substring(0, i);
143 
144             LOG.debug("Updating parent : " + substr);
145             SettingsKey key = new SettingsKey(substr); // simple constructor
146             Object o = map.get(key);
147             // Create a provision node for a future parent.
148             if (o == null) {
149                 // System.out.println("No parent "+substr+" found. Creating ProvisionNode.");
150                 ProvisionNode pn = new ProvisionNode(settings);
151                 map.put(key, pn);
152             } else if (o instanceof Settings) {
153                 parentFound = true;
154                 settings.setParent((Settings) o);
155                 // System.out.println("Linking " + cat.name + " -> " + ((Category) o).name);
156                 break; // no need to update the ancestors of the closest ancestor
157             } else if (o instanceof ProvisionNode) {
158                 ((ProvisionNode) o).addElement(settings);
159             } else {
160                 Exception e = new IllegalStateException("unexpected object type " + o.getClass() + " in ht.");
161                 e.printStackTrace();
162             }
163         }
164         // If we could not find any existing parents, then link with root.
165         if (!parentFound)
166             settings.setParent(root);
167     }

169     /**
170      * We update the links for all the children that placed themselves in the provision node 'pn'.
171      * The second argument 'cat' is a reference for the newly created Logger, parent of all the
172      * children in 'pn'
173      * 
174      * We loop on all the children 'c' in 'pn':
175      * 
176      * If the child 'c' has been already linked to a child of 'cat' then there is no need to update
177      * 'c'.
178      * 
179      * Otherwise, we set cat's parent field to c's parent and set c's parent field to cat.
180      * 
181      */
182     final private void updateChildren(ProvisionNode pn, Settings settings) {
183         // System.out.println("updateChildren called for " + logger.name);
184         final int last = pn.size();
185 
186         for (int i = 0; i < last; i++) {
187             Settings s = (Settings) pn.elementAt(i);
188             // System.out.println("Updating child " +p.name);
189 
190             // Unless this child already points to a correct (lower) parent,
191             // make cat.parent point to l.parent and l.parent to cat.
192             if (!s.getParent().getName().startsWith(settings.getName())) {
193                 settings.setParent(s.getParent());
194                 s.setParent(settings);
195             }
196         }
197     }
