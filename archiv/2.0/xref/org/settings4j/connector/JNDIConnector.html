53      /** {@inheritDoc} */
54      public byte[] getContent(final String key) { 
55          final Object obj = lookupInContext(key);
56  
57          // if obj is a String and an Object resolver is available
58          // obj could be a Path.
59          if (obj instanceof String && getContentResolver() != null) {
60              final byte[] content = getContentResolver().getContent((String) obj);
61              if (content != null) {
62                  return content;
63              }
64          }
65  
66          if (obj instanceof byte[]) {
67              return (byte[]) obj;
68          }
69          return null;
70      }
71  
72      /** {@inheritDoc} */
73      public Object getObject(final String key) {
74          final Object obj = lookupInContext(key);
75  
76          // if obj is a String and an Object resolver is available
77          // obj could be a Path to a XML who can be converted to an Object.
78          if (obj instanceof String && getObjectResolver() != null) {
79              final Object convertedObject = getObjectResolver().getObject((String) obj, getContentResolver());
80              if (convertedObject != null) {
81                  return convertedObject;
82              }
83          }
84  
85          return obj;
86      }
87  
88      /** {@inheritDoc} */
89      public String getString(final String key) {
90          final Object obj = lookupInContext(key);
91          try {
92              return (String) obj;
93          } catch (final ClassCastException e) {
94              LOG.warn("Wrong Type: {} for Key: {}", obj.getClass().getName(), key);
95              LOG.debug(e.getMessage(), e);
96              return null;
97          }
98      }

100     /**
101      * Set or replace a new Value for the given key.<br />
102      * If set or replace a value is not possible because of a readonly JNDI-Context, then
103      * {@link Constants#SETTING_NOT_POSSIBLE} must be returned. If set or replace was successful, then
104      * {@link Constants#SETTING_SUCCESS} must be returned.
105      * 
106      * @param key the Key for the configuration-property. e.g.: "com/mycompany/myapp/myParameterKey"
107      * @param value the new Object-Value for the given key
108      * @return Returns {@link Constants#SETTING_SUCCESS} or {@link Constants#SETTING_NOT_POSSIBLE}
109      */
110     public int setObject(final String key, final Object value) {
111         return rebindToContext(key, value);
112     }

114     private InitialContext getJNDIContext() throws NamingException {
115         InitialContext initialContext;
116 
117         if (StringUtils.isEmpty(this.providerUrl) && StringUtils.isEmpty(this.initialContextFactory)
118             && StringUtils.isEmpty(this.urlPkgPrefixes)) {
119 
120             initialContext = new InitialContext();
121         } else {
122             final Properties prop = new Properties();
123             prop.put(Context.PROVIDER_URL, this.providerUrl);
124             prop.put(Context.INITIAL_CONTEXT_FACTORY, this.initialContextFactory);
125             prop.put(Context.URL_PKG_PREFIXES, this.urlPkgPrefixes);
126             initialContext = new InitialContext(prop);
127         }
128 
129         return initialContext;
130     }

132     /**
133      * check if a JNDI context is available and sets the internal Flag setIsJNDIAvailable(Boolean).
134      * <p>
135      * If the internal Flag IsJNDIAvailable is <code>False</code> this Connector is disabled.
136      * 
137      * @return true if a JNDI Context could be initialized.
138      */
139     public boolean isJNDIAvailable() {
140         if (this.isJNDIAvailable == null) {
141             try {
142                 getJNDIContext().lookup(getContextPathPrefix());
143                 LOG.debug("JNDI Context is available.");
144                 this.isJNDIAvailable = Boolean.TRUE;
145             } catch (final NoInitialContextException e) {
146                 LOG.info("No JNDI Context available! JNDIConnector will be disabled: {}", e.getMessage());
147                 this.isJNDIAvailable = Boolean.FALSE;
148             } catch (final NamingException e) {
149                 LOG.info("JNDI Context is available but {}", e.getMessage());
150                 LOG.debug("NamingException in isJNDIAvailable: " + e.getMessage(), e);
151                 this.isJNDIAvailable = Boolean.TRUE;
152             }
153         }
154 
155         return this.isJNDIAvailable.booleanValue();
156     }

170     private Object lookupInContext(final String key) {
171         if (!isJNDIAvailable()) {
172             return null;
173         }
174         final String normalizedKey = normalizeKey(key);
175         InitialContext ctx = null;
176         try {
177             Object result;
178             ctx = getJNDIContext();
179             result = ctx.lookup(normalizedKey);
180             return result;
181         } catch (final NoInitialContextException e) {
182             LOG.info("Maybe no JNDI-Context available.");
183             LOG.debug(e.getMessage(), e);
184         } catch (final NamingException e) {
185             LOG.debug("cannot lookup key: " + key + " (" + normalizedKey + ")", e);
186         } finally {
187             if (ctx != null) {
188                 try {
189                     ctx.close();
190                 } catch (final NamingException e) {
191                     LOG.info("cannot close context: " + key + " (" + normalizedKey + ")", e);
192                 }
193             }
194         }
195         return null;
196     }

198     /**
199      * @param key the JNDI-Key (will be normalized: add contextPathPrefix, replace '\' with '/').
200      * @param value the JNDI-Value.
201      * @return Constants.SETTING_NOT_POSSIBLE if the JNDI Context ist readonly.
202      */
203     public int rebindToContext(final String key, final Object value) {
204         // don't do a check, but use the result if a check was done.
205         if (BooleanUtils.isFalse(this.isJNDIAvailable)) {
206             // only if isJNDIAvailable() was called an evaluated to false.
207             return Constants.SETTING_NOT_POSSIBLE;
208         }
209         final String normalizedKey = normalizeKey(key);
210         LOG.debug("Try to rebind Key '{}' ({}) with value: {}", key, normalizedKey, value);
211 
212         InitialContext ctx = null;
213         int result = Constants.SETTING_NOT_POSSIBLE;
214         try {
215             ctx = getJNDIContext();
216             createParentContext(ctx, normalizedKey);
217             ctx.rebind(normalizedKey, value);
218             result = Constants.SETTING_SUCCESS;
219         } catch (final NoInitialContextException e) {
220             LOG.info("Maybe no JNDI-Context available.");
221             LOG.debug(e.getMessage(), e);
222         } catch (final NamingException e) {
223             // the JNDI-Context from TOMCAT is readonly
224             // if you try to write it, The following Exception will be thrown:
225             // javax.naming.NamingException: Context is read only
226             LOG.info("cannot bind key: '{}' ({}). {}", key, normalizedKey, e.getMessage());
227             if (LOG.isDebugEnabled()) {
228                 LOG.debug("cannot bind key: " + key + " (" + normalizedKey + ")", e);
229             }
230         } finally {
231             if (ctx != null) {
232                 try {
233                     ctx.close();
234                 } catch (final NamingException e) {
235                     LOG.info("cannot close context: " + key + " (" + normalizedKey + ")", e);
236                 }
237             }
238         }
239         return result;
240     }

242     private static void createParentContext(final Context ctx, final String key) throws NamingException {
243         // here we need to break by the specified delimiter
244 
245         LOG.debug("createParentContext: {}", key);
246         
247         final String[] path = key.split("/");
248 
249         final int lastIndex = path.length - 1;
250 
251         Context tmpCtx = ctx;
252 
253         for (int i = 0; i < lastIndex; i++) {
254             Object obj = null;
255             try {
256                 obj = tmpCtx.lookup(path[i]);
257             } catch (final NameNotFoundException e) {
258                 LOG.debug("obj is null and subcontext will be generated: {}", path[i]);
259             }
260 
261             if (obj == null) {
262                 tmpCtx = tmpCtx.createSubcontext(path[i]);
263                 LOG.debug("createSubcontext: {}", path[i]);
264             } else if (obj instanceof Context) {
265                 tmpCtx = (Context) obj;
266             } else {
267                 throw new RuntimeException("Illegal node/branch clash. At branch value '" + path[i]
268                     + "' an Object was found: " + obj);
269             }
270         }
271     }

273     private String normalizeKey(final String key) {
274         if (key == null) {
275             return null;
276         }
277         String normalizeKey = key;
278         if (normalizeKey.startsWith(this.contextPathPrefix)) {
279             return normalizeKey;
280         }
281 
282         normalizeKey = normalizeKey.replace('\\', '/');
283 
284         if (normalizeKey.startsWith("/")) {
285             normalizeKey = normalizeKey.substring(1);
286         }
287         return this.contextPathPrefix + normalizeKey;
288     }
