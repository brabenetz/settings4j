137     private void cachingTest(final String key, final JavaXMLBeansObjectResolver objectResolver) throws Exception {
138         final ContentResolver contentResolver = createUnionContentResolver();
139 
140         // read from classpath
141         final Map result = (Map) objectResolver.getObject(key, contentResolver);
142         assertEquals("blablablaNEUblablabla", result.get("irgendwasNeues"));
143         final Object liste = result.get("liste");
144         assertNotNull(liste);
145         assertTrue(liste instanceof List);
146         assertEquals(1, ((List) liste).size());
147         assertEquals("testValue1", ((List) liste).get(0));
148 
149         Map result2 = (Map) objectResolver.getObject(key, contentResolver);
150         // this Object is cached! The two Objects must be the same.
151         assertTrue(result == result2);
152         assertEquals(result2.get("irgendwasNeues"), result.get("irgendwasNeues"));
153 
154         final Map testData = new HashMap();
155         final List testList = new ArrayList();
156         final String testValue1 = "testValue1";
157         final String testValue2 = "testValue2";
158         final String testValue3 = "testValue3";
159         final String testValue4 = "testValue4";
160         testList.add(testValue1);
161         testList.add(testValue2);
162         testList.add(testValue3);
163         testList.add(testValue4);
164         testData.put("irgendwas", "blablablablablabla");
165         testData.put("liste", testList);
166 
167         // copy properties for a Temp-Key. This is required for parsing (read/write) of Objects
168         byte[] content = contentResolver.getContent(key + ".properties");
169         setContent(key + "temp.properties", content);
170 
171         // save Object to a Temp-Key
172         setObject(key + "temp", testData);
173 
174         // Copy the writen Object without Objectreolver from Temp-Key to real Key
175         content = contentResolver.getContent(key + "temp");
176         setContent(key, content);
177 
178 
179         // this Object is cached, and the ObjectResolver doesn't know that the content was changed from contentResolver!
180         // The OLD Object must be returned.
181         result2 = (Map) objectResolver.getObject(key, contentResolver);
182         assertTrue(result2 == result);
183         assertEquals(result2.get("irgendwasNeues"), result.get("irgendwasNeues"));
184 
185     }

210     private byte[] objectToContent(final Object value) {
211         final ByteArrayOutputStream byteArrayOutputStream = new ByteArrayOutputStream();
212 
213         final XMLEncoder encoder = new XMLEncoder(byteArrayOutputStream);
214         encoder.setExceptionListener(new LogEncoderExceptionListener(value));
215         if (LOG.isDebugEnabled()) {
216             LOG.debug("START Writing Object " + value.getClass().getName() + " with XMLEncoder");
217         }
218         encoder.writeObject(value);
219         if (LOG.isDebugEnabled()) {
220             LOG.debug("FINISH Writing Object " + value.getClass().getName() + " with XMLEncoder");
221         }
222         encoder.flush();
223         encoder.close();
224         return byteArrayOutputStream.toByteArray();
225     }

227     /**
228      * Log out Exception during Encoding a byte[] to an Object<br />
229      * <br />
230      * Example:<br />
231      * The {@link org.springframework.jdbc.datasource.AbstractDataSource} Object<br />
232      * hast Getter and Setter for "logWriter" who throws per default an {@link UnsupportedOperationException}.<br />
233      * 
234      * @author hbrabenetz
235      */
236     private class LogEncoderExceptionListener implements ExceptionListener {
237 
238         private final Object obj;
239 
240         public LogEncoderExceptionListener(final Object obj) {
241             super();
242             this.obj = obj;
243         }
244 
245         /** {@inheritDoc} */
246         public void exceptionThrown(final Exception e) {
247             LOG.warn("Ignore error on encoding Object from type: " + this.obj.getClass().getName() + "! "
248                 + e.getClass().getName() + ": '" + e.getMessage() + "'. Set Loglevel DEBUG for more informations.");
249             if (LOG.isDebugEnabled()) {
250                 LOG.debug(e.getMessage(), e);
251             }
252         }
