17  package org.settings4j.settings;
18  
19  import java.net.URL;
20  
21  import org.settings4j.Settings4jFactory;
22  import org.settings4j.Settings4jInstance;
23  import org.settings4j.Settings4jRepository;
24  import org.settings4j.config.DOMConfigurator;
25  import org.settings4j.contentresolver.ClasspathContentResolver;
26  import org.settings4j.settings.nop.NOPSettingsRepository;
27  
28  /**
29   * managed The {@link Settings4jRepository} . This {@link Settings4jRepository} is used to store the configuration from
30   * the settings4j.xml.
31   * 
32   * @author Harald.Brabenetz
33   */
34  public final class SettingsManager {
35  
36      /** General Logger for this Class. */
37      private static final org.apache.commons.logging.Log LOG = org.apache.commons.logging.LogFactory
38          .getLog(SettingsManager.class);
39  
40      /**
41       * The Classpath-Location of the settings4j.xml who i readed by default. <br />
42       * settings4j.xml
43       */
44      public static final String DEFAULT_XML_CONFIGURATION_FILE = "settings4j.xml";
45  
46      /**
47       * The fallback settings4j.xml who is used if not settings4j.xml where found. <br />
48       * org/settings4j/config/defaultsettings4j.xml
49       */
50      public static final String DEFAULT_FALLBACK_CONFIGURATION_FILE = "org/settings4j/config/defaultsettings4j.xml";
51  
52      /**
53       * The internal default Settings4j Repository where all {@link org.settings4j.Settings4j} are stored.
54       */
55      private static Settings4jRepository settingsRepository;
56      static {
57  
58          /**
59           * The root settings names itself as "root". However, the root settings cannot be retrieved by name.
60           */
61          settingsRepository = new DefaultSettingsRepository();
62  
63          // read XML default Configuration to configure the repository
64          final URL url = ClasspathContentResolver.getResource(DEFAULT_XML_CONFIGURATION_FILE);
65  
66          // If we have a non-null url, then delegate the rest of the
67          // configuration to the DOMConfigurator.configure method.
68          if (url != null) {
69              LOG.debug("Using URL [" + url + "] for automatic settings4j configuration.");
70              try {
71                  DOMConfigurator.configure(url, settingsRepository);
72              } catch (final NoClassDefFoundError e) {
73                  LOG.warn("Error during default initialization", e);
74              }
75          } else {
76              LOG.debug("Could not find resource: [" + DEFAULT_XML_CONFIGURATION_FILE + "].");
77          }
78      }
79  
80      /** Hide Constructor, Utility Pattern. */
81      private SettingsManager() {
82          super();
83      }
84  
85      /**
86       * The internal default Settings4j Repository where all {@link org.settings4j.Settings4j} are stored.
87       * 
88       * @return The Settings4j Repository, Returns NEVER null put maybe a {@link NOPSettingsRepository}.
89       */
90      public static Settings4jRepository getSettingsRepository() {
91          if (settingsRepository == null) {
92              settingsRepository = new NOPSettingsRepository();
93              LOG.error("SettingsManager.settingsRepository was null likely due to error in class reloading, " //
94                  + "using the NOPSettingsRepository.");
95          }
96          return settingsRepository;
97      }
98  
99      /**
100      * Retrieve the appropriate {@link org.settings4j.Settings4j} instance.
101      * 
102      * @return the appropriate {@link org.settings4j.Settings4j} instance.
103      */
104     public static Settings4jInstance getSettings() {
105         initializeRepositoryIfNecessary();
106         // Delegate the actual manufacturing of the settings to the settings repository.
107         return getSettingsRepository().getSettings();
108     }
109 
110     /**
111      * Retrieve the appropriate {@link org.settings4j.Settings4j} instance or create it with the give factory if doesn'r
112      * already exist.
113      * 
114      * @param factory The factory to create a {@link Settings4jInstance}.
115      * @return the appropriate {@link org.settings4j.Settings4j} instance.
116      */
117     public static Settings4jInstance getSettings(final Settings4jFactory factory) {
118         initializeRepositoryIfNecessary();
119         // Delegate the actual manufacturing of the settings to the settings repository.
120         return getSettingsRepository().getSettings(factory);
121     }
122 
123     /**
124      * Check if the repository must be configured with the defaul fallback settings4j.xml.
125      */
126     private static void initializeRepositoryIfNecessary() {
127         if (getSettingsRepository().getConnectorCount() == 0) {
128             // No connectors in hierarchy, warn user and add default-configuration.
129             LOG.info("The settings4j will be configured with the default-fallback-config: "
130                 + SettingsManager.DEFAULT_FALLBACK_CONFIGURATION_FILE);
131 
132             final URL url = ClasspathContentResolver.getResource(SettingsManager.DEFAULT_FALLBACK_CONFIGURATION_FILE);
133 
134             // If we have a non-null url, then delegate the rest of the
135             // configuration to the DOMConfigurator.configure method.
136             if (url != null) {
137                 LOG.debug("Using URL [" + url + "] for automatic settings4j fallback configuration.");
138                 try {
139                     DOMConfigurator.configure(url, getSettingsRepository());
140                 } catch (final NoClassDefFoundError e) {
141                     LOG.warn("Error during default fallback initialization", e);
142                 }
143             } else {
144                 LOG.fatal("Could not find resource: [" + SettingsManager.DEFAULT_FALLBACK_CONFIGURATION_FILE + "].");
145             }
146         }
147     }
