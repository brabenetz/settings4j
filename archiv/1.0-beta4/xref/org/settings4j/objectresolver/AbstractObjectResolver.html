18  package org.settings4j.objectresolver;
19  
20  import java.io.ByteArrayInputStream;
21  import java.io.IOException;
22  import java.util.HashMap;
23  import java.util.Map;
24  import java.util.Properties;
25  
26  import org.apache.commons.lang.StringUtils;
27  import org.settings4j.ContentResolver;
28  import org.settings4j.ObjectResolver;
29  
30  /**
31   * Basic Connector implementations like getter and Setter of contentResolver, objectResolver.
32   * <p>
33   * 
34   * @author Harald.Brabenetz
35   *
36   */
37  public abstract class AbstractObjectResolver implements ObjectResolver {
38  
39      /** General Logger for this Class. */
40      private static final org.apache.commons.logging.Log LOG = org.apache.commons.logging.LogFactory
41          .getLog(AbstractObjectResolver.class);
42  
43      /** The Key which ObjectResolver implementation should be used. */
44      public static final String PROP_OBJECT_RESOLVER_KEY = "objectResolverKey";
45      /** The Key if the found Object should be cached (this will handle the Object as singleton). */
46      public static final String PROP_CACHED = "cached";
47  
48      private String propertySuffix = ".properties";
49  
50      private final Map cachedObjects = new HashMap();
51  
52      private boolean cached = false;
53  
54      /** {@inheritDoc} */
55      public void addObjectResolver(final ObjectResolver objectResolver) {
56          throw new UnsupportedOperationException(this.getClass().getName() + " cannot add other ObjectResolvers");
57      }
58  
59      /** {@inheritDoc} */
60      public Object getObject(final String key, final ContentResolver contentResolver) {
61  
62          Object result = this.cachedObjects.get(key);
63          if (result != null) {
64              return result;
65          }
66  
67  
68          final byte[] content = contentResolver.getContent(key);
69          if (content != null) {
70              final Properties properties = getObjectProperties(key, contentResolver);
71              if (properties != null) {
72                  final String propObjectResolverKey = properties.getProperty(PROP_OBJECT_RESOLVER_KEY);
73                  final String propCached = properties.getProperty(PROP_CACHED);
74                  if (StringUtils.isEmpty(propObjectResolverKey)) {
75                      LOG.warn("The property-File for Key '" + key //
76                          + "' doesn't have the required Property '" + PROP_OBJECT_RESOLVER_KEY + "'");
77                      return null;
78                  }
79  
80                  if (getObjectResolverKey().equals(propObjectResolverKey)) {
81                      result = contentToObject(key, properties, content, contentResolver);
82                      if (result != null) {
83                          if ("true".equalsIgnoreCase(propCached) || (propCached == null && isCached())) {
84                              this.cachedObjects.put(key, result);
85                          }
86                          return result;
87                      }
88                  }
89              }
90          }
91          return null;
92      }
93  
94      /**
95       * To get the additional Properties for the given byte[] content can be overwriten by subclasses.
96       * <p>
97       * The default implementation reads the PropertyFile from key + ".properties".
98       * If no property where found, or an error occurs, this method return null.
99       * 
100      * @param key the key of the byte[] Content to convert.
101      * @param contentResolver the ContentResolver to read the additional Property-File.
102      * @return the parsed {@link Properties} Object or null if an error occurred.
103      */
104     protected Properties getObjectProperties(final String key, final ContentResolver contentResolver) {
105 
106         final byte[] propertyContent = contentResolver.getContent(key + this.propertySuffix);
107         if (propertyContent == null) {
108             return null;
109         }
110         // else
111         final Properties properties = new Properties();
112         try {
113             properties.load(new ByteArrayInputStream(propertyContent));
114         } catch (final IOException e) {
115             LOG.error(e.getMessage(), e);
116             return null;
117         }
118         return properties;
119         
120     }
121 
122     public String getPropertySuffix() {
123         return this.propertySuffix;
124     }
125 
126     public void setPropertySuffix(final String propertySuffix) {
127         this.propertySuffix = propertySuffix;
128     }
129 
130     protected String getObjectResolverKey() {
131         return this.getClass().getName();
132     }
133 
134     /**
135      * Method to convert the given content-File to an Object must be implemented by SubClasses.
136      * <p>
137      * 
138      * @param key The Original Key of the Object
139      * @param properties The Property-File which where Found under key + ".properties"
140      * @param content The byte[] Content to convert.
141      * @param contentResolver the contentResolver to get possible additional content Files.
142      * @return the parsed Object from the byte[] Content.
143      */
144     protected abstract Object contentToObject(String key, Properties properties, byte[] content,
145             ContentResolver contentResolver);
146 
147     public boolean isCached() {
148         return this.cached;
149     }
150 
151     public void setCached(final boolean cached) {
152         this.cached = cached;
153     }
154 }
