142     /**
143      * Sets a parameter based from configuration file content.
144      * 
145      * @param elem param element, may not be null.
146      * @param propSetter property setter, may not be null.
147      * @param props properties
148      */
149     private void setParameter(final Element elem, final Object bean, final Connector[] connectors) {
150         final String name = elem.getAttribute(NAME_ATTR);
151         final String valueStr = (elem.getAttribute(VALUE_ATTR));
152         try {
153             final PropertyDescriptor propertyDescriptor = PropertyUtils.getPropertyDescriptor(bean, name);
154             final Method setter = PropertyUtils.getWriteMethod(propertyDescriptor);
155             Object value;
156             if (connectors != null) {
157                 value = subst(valueStr, connectors, setter.getParameterTypes()[0]);
158             } else {
159                 value = subst(valueStr, null, setter.getParameterTypes()[0]);
160             }
161             PropertyUtils.setProperty(bean, name, value);
162         } catch (final IllegalAccessException e) {
163             LOG.warn("Cannnot set Property: " + name, e);
164         } catch (final InvocationTargetException e) {
165             LOG.warn("Cannnot set Property: " + name, e);
166         } catch (final NoSuchMethodException e) {
167             LOG.warn("Cannnot set Property: " + name, e);
168         }
169     }

183     /**
184      * @param url The location of the configuration file.
185      */
186     public void doConfigure(final URL url) {
187         final ParseAction action = new ParseAction() {
188 
189             public Document parse(final DocumentBuilder parser) throws SAXException, IOException {
190                 return parser.parse(url.toString());
191             }
192 
193             public String toString() {
194                 return "url [" + url.toString() + "]";
195             }
196         };
197         doConfigure(action);
198     }

200     private void doConfigure(final ParseAction action) throws FactoryConfigurationError {
201         DocumentBuilderFactory dbf = null;
202         try {
203             LOG.debug("System property is :" + new SystemPropertyConnector().getString(DOCUMENT_BUILDER_FACTORY_KEY));
204             dbf = DocumentBuilderFactory.newInstance();
205             LOG.debug("Standard DocumentBuilderFactory search succeded.");
206             LOG.debug("DocumentBuilderFactory is: " + dbf.getClass().getName());
207         } catch (final FactoryConfigurationError fce) {
208             final Exception e = fce.getException();
209             LOG.debug("Could not instantiate a DocumentBuilderFactory.", e);
210             throw fce;
211         }
212 
213         try {
214             dbf.setValidating(true);
215 
216             final DocumentBuilder docBuilder = dbf.newDocumentBuilder();
217 
218             docBuilder.setErrorHandler(new SAXErrorHandler());
219             docBuilder.setEntityResolver(new Settings4jEntityResolver());
220 
221             final Document doc = action.parse(docBuilder);
222             parse(doc.getDocumentElement());
223         } catch (final Exception e) {
224             // I know this is miserable...
225             LOG.error("Could not parse " + action.toString() + ".", e);
226         }
227     }

229     /**
230      * Used internally to configure the settings4j framework by parsing a DOM tree of XML elements based on <a
231      * href="doc-files/settings4j.dtd">settings4j.dtd</a>.
232      * 
233      * @param element The XML {@link #CONFIGURATION_TAG} Element.
234      */
235     protected void parse(final Element element) {
236 
237         final String rootElementName = element.getTagName();
238 
239         if (!rootElementName.equals(CONFIGURATION_TAG)) {
240             LOG.error("DOM element is - not a <" + CONFIGURATION_TAG + "> element.");
241             return;
242         }
243 
244         final Settings4jInstance root = this.repository.getSettings();
245         // settings configuration needs to be atomic
246         synchronized (root) {
247             parseChildrenOfSettingsElement(element, root);
248         }
249     }

251     /**
252      * Used internally to parse an Filter element.
253      * 
254      * @param filterElement the XML-Element for Filter.
255      * @return the Filter-Object
256      */
257     protected Filter parseFilter(final Element filterElement) {
258 
259         Filter filter;
260 
261 
262         String className = filterElement.getAttribute(CLASS_ATTR);
263         if (StringUtils.isEmpty(className)) {
264             className = DefaultFilter.class.getName();
265         }
266 
267         LOG.debug("Desired Connector class: [" + className + "]");
268         try {
269             final Class clazz = loadClass(className);
270             final Constructor constructor = clazz.getConstructor(NO_PARAM);
271             filter = (Filter) constructor.newInstance(null);
272         } catch (final Exception oops) {
273             LOG.error("Could not retrieve connector [filter: " + className + "]. Reported error follows.", oops);
274             return null;
275         }
276 
277         final NodeList children = filterElement.getChildNodes();
278         final int length = children.getLength();
279 
280         for (int loop = 0; loop < length; loop++) {
281             final Node currentNode = children.item(loop);
282 
283             if (currentNode.getNodeType() == Node.ELEMENT_NODE) {
284                 final Element currentElement = (Element) currentNode;
285                 final String tagName = currentElement.getTagName();
286 
287                 if (tagName.equals(INCLUDE_TAG)) {
288                     final Element includeTag = (Element) currentNode;
289                     final String patteren = includeTag.getAttribute(PATTERN_ATTR);
290                     filter.addInclude(patteren);
291 
292                 } else if (tagName.equals(EXCLUDE_TAG)) {
293                     final Element excludeTag = (Element) currentNode;
294                     final String patteren = excludeTag.getAttribute(PATTERN_ATTR);
295                     filter.addExclude(patteren);
296                 } else {
297                     quietParseUnrecognizedElement(filter, currentElement);
298                 }
299             }
300         }
301 
302         return filter;
303     }

305     /**
306      * Used internally to parse an connector element.
307      * 
308      * @param connectorElement The XML Connector Element
309      * @return the Connector Object.
310      */
311     protected Connector parseConnector(final Element connectorElement) {
312         final String connectorName = connectorElement.getAttribute(NAME_ATTR);
313 
314         Connector connector;
315 
316         final String className = connectorElement.getAttribute(CLASS_ATTR);
317 
318 
319         LOG.debug("Desired Connector class: [" + className + ']');
320         try {
321             final Class clazz = loadClass(className);
322             final Constructor constructor = clazz.getConstructor(NO_PARAM);
323             connector = (Connector) constructor.newInstance(null);
324         } catch (final Exception oops) {
325             LOG.error("Could not retrieve connector [" + connectorName + "]. Reported error follows.", oops);
326             return null;
327         }
328 
329         connector.setName(connectorName);
330 
331         final Connector[] subConnectors = getConnectors(connectorElement);
332         for (int i = 0; i < subConnectors.length; i++) {
333             connector.addConnector(subConnectors[i]);
334         }
335 
336         Filter filter = null;
337         // Setting up a connector needs to be an atomic operation, in order
338         // to protect potential setXXX operations while connector
339         // configuration is in progress.
340         synchronized (connector) {
341 
342             final NodeList children = connectorElement.getChildNodes();
343             final int length = children.getLength();
344 
345             for (int loop = 0; loop < length; loop++) {
346                 final Node currentNode = children.item(loop);
347 
348                 if (currentNode.getNodeType() == Node.ELEMENT_NODE) {
349                     final Element currentElement = (Element) currentNode;
350                     final String tagName = currentElement.getTagName();
351 
352                     if (tagName.equals(CONTENT_RESOLVER_REF_TAG)) {
353                         final Element contentResolverRef = (Element) currentNode;
354                         final ContentResolver contentResolver = findContentResolverByReference(contentResolverRef);
355                         connector.setContentResolver(contentResolver);
356 
357                     } else if (tagName.equals(OBJECT_RESOLVER_REF_TAG)) {
358                         final Element objectResolverRef = (Element) currentNode;
359                         final ObjectResolver objectResolver = findObjectResolverByReference(objectResolverRef);
360                         connector.setObjectResolver(objectResolver);
361 
362                     } else if (tagName.equals(FILTER_TAG)) {
363                         final Element filterElement = (Element) currentNode;
364                         filter = parseFilter(filterElement);
365                     } else if (tagName.equals(PARAM_TAG)) {
366                         setParameter(currentElement, connector, subConnectors);
367                     } else {
368                         quietParseUnrecognizedElement(connector, currentElement);
369                     }
370                 }
371             }
372 
373             final Boolean cached = (Boolean) subst(connectorElement.getAttribute(CACHED_ATTR), subConnectors,
374                 Boolean.class);
375             if (cached != null && cached.booleanValue()) {
376                 connector = new CachedConnectorWrapper(connector);
377             }
378 
379             if (filter != null) {
380                 connector = new FilteredConnectorWrapper(connector, filter);
381             }
382 
383             // initial the connector
384             connector.init();
385         }
386         return connector;
387     }

389     /**
390      * Only logs out unrecognized Elements.
391      * 
392      * @param instance instance, may be null.
393      * @param element element, may not be null.
394      */
395     private static void quietParseUnrecognizedElement(final Object instance, final Element element) {
396         String elementName = "UNKNOWN";
397         String instanceClassName = "UNKNOWN";
398 
399         try {
400             elementName = element.getNodeName();
401             instanceClassName = instance.getClass().getName();
402         } catch (final Exception e) {
403             LOG.warn("Error in quietParseUnrecognizedElement(): " + e.getMessage());
404             if (LOG.isDebugEnabled()) {
405                 LOG.debug(e.getMessage(), e);
406             }
407         }
408         LOG.warn("Unrecognized Element will be ignored: " + elementName + " for Instance: " + instanceClassName);
409     }

411     /**
412      * Return all referenced connectors from Child-Nodes {@link #CONNECTOR_REF_TAG}.
413      * 
414      * @param connectorsElement The XML Connectors Element
415      * @return the Connectors Objects as Array.
416      */
417     protected Connector[] getConnectors(final Element connectorsElement) {
418         final List connectors = new ArrayList();
419 
420         final NodeList children = connectorsElement.getChildNodes();
421         final int length = children.getLength();
422 
423         for (int loop = 0; loop < length; loop++) {
424             final Node currentNode = children.item(loop);
425 
426             if (currentNode.getNodeType() == Node.ELEMENT_NODE) {
427                 final Element currentElement = (Element) currentNode;
428                 final String tagName = currentElement.getTagName();
429 
430                 if (tagName.equals(CONNECTOR_REF_TAG)) {
431                     final Element connectorRef = (Element) currentNode;
432                     final Connector connector = findConnectorByReference(connectorRef);
433                     connectors.add(connector);
434                 }
435             }
436         }
437 
438         return (Connector[]) connectors.toArray(new Connector[connectors.size()]);
439     }

442     /**
443      * Used internally to parse the children of a settings element.
444      * 
445      * @param settingsElement The XML Settings Element
446      * @param settings _The Settings Object do apply the values.
447      */
448     protected void parseChildrenOfSettingsElement(final Element settingsElement, final Settings4jInstance settings) {
449 
450         Node currentNode = null;
451         Element currentElement = null;
452         String tagName = null;
453 
454         // Remove all existing appenders from settings. They will be
455         // reconstructed if need be.
456         settings.removeAllConnectors();
457 
458         // first parse Connectors (are needed to parse param Tags
459         final NodeList connectorElements = settingsElement.getElementsByTagName(CONNECTOR_TAG);
460         int length = connectorElements.getLength();
461         for (int i = 0; i < length; i++) {
462             currentNode = connectorElements.item(i);
463             currentElement = (Element) currentNode;
464 
465             final Connector connector = parseConnector(currentElement);
466             if (connector != null) {
467                 this.connectorBag.put(connector.getName(), connector);
468                 settings.addConnector(connector);
469             }
470         }
471 
472         final List list = settings.getConnectors();
473         final Connector[] connectors = (Connector[]) list.toArray(new Connector[list.size()]);
474 
475         final NodeList children = settingsElement.getChildNodes();
476 
477         // Now parse other Tags like PARAM or MAPPING
478         length = children.getLength();
479         for (int i = 0; i < length; i++) {
480             currentNode = children.item(i);
481             if (currentNode.getNodeType() == Node.ELEMENT_NODE) {
482                 currentElement = (Element) currentNode;
483                 tagName = currentElement.getTagName();
484 
485                 if (tagName.equals(MAPPING_TAG)) {
486                     final Map mapping = parseMapping(currentElement);
487                     if (mapping != null) {
488                         settings.setMapping(mapping);
489                     }
490                 } else if (tagName.equals(PARAM_TAG)) {
491                     setParameter(currentElement, settings, connectors);
492                 } else if (tagName.equals(CONNECTOR_TAG)) {
493                     LOG.trace("CONNECTOR_TAG already parsed");
494                 } else if (tagName.equals(CONTENT_RESOLVER_TAG)) {
495                     LOG.trace("CONTENT_RESOLVER_TAG will be parsed on the maned");
496                 } else if (tagName.equals(OBJECT_RESOLVER_TAG)) {
497                     LOG.trace("OBJECT_RESOLVER_TAG will be parsed on the maned");
498                 } else {
499                     quietParseUnrecognizedElement(settings, currentElement);
500                 }
501             }
502         }
503     }

506     /**
507      * Used internally to parse connectors by IDREF name.
508      * 
509      * @param doc The whole XML configuration.
510      * @param connectorName The Connector-Name, to search for.
511      * @return the Connector instance.
512      */
513     protected Connector findConnectorByName(final Document doc, final String connectorName) {
514         Connector connector = (Connector) this.connectorBag.get(connectorName);
515 
516         if (connector != null) {
517             return connector;
518         }
519         // else
520         final Element element = getElementByNameAttr(doc, connectorName, "connector");
521 
522         if (element == null) {
523             LOG.error("No connector named [" + connectorName + "] could be found.");
524             return null;
525         }
526         // else
527         connector = parseConnector(element);
528         this.connectorBag.put(connectorName, connector);
529         return connector;
530     }

545     /**
546      * Used internally to parse an objectResolver element.
547      * 
548      * @param objectResolverElement The XML Object resolver Element.
549      * @return The ObjectResolver instance.
550      */
551     protected ObjectResolver parseObjectResolver(final Element objectResolverElement) {
552         final String objectResolverName = objectResolverElement.getAttribute(NAME_ATTR);
553 
554         ObjectResolver objectResolver;
555 
556         final String className = objectResolverElement.getAttribute(CLASS_ATTR);
557 
558 
559         LOG.debug("Desired ObjectResolver class: [" + className + ']');
560         try {
561             final Class clazz = loadClass(className);
562             final Constructor constructor = clazz.getConstructor(NO_PARAM);
563             objectResolver = (ObjectResolver) constructor.newInstance(null);
564         } catch (final Exception oops) {
565             LOG.error("Could not retrieve objectResolver [" + objectResolverName + "]. Reported error follows.", oops);
566             return null;
567         } catch (final NoClassDefFoundError e) {
568             LOG.warn("The ObjectResolver '" + objectResolverName
569                 + "' cannot be created. There are not all required Libraries inside the Classpath: " + e.getMessage(),
570                 e);
571             return null;
572         }
573 
574         // get connectors for ExpressionLanguage validation
575         final Connector[] connectors = getConnectors(objectResolverElement);
576 
577         Filter filter = null;
578         // Setting up a objectResolver needs to be an atomic operation, in order
579         // to protect potential settings operations while settings
580         // configuration is in progress.
581         synchronized (objectResolver) {
582 
583             final NodeList children = objectResolverElement.getChildNodes();
584             final int length = children.getLength();
585 
586             for (int loop = 0; loop < length; loop++) {
587                 final Node currentNode = children.item(loop);
588 
589                 if (currentNode.getNodeType() == Node.ELEMENT_NODE) {
590                     final Element currentElement = (Element) currentNode;
591                     final String tagName = currentElement.getTagName();
592 
593                     if (tagName.equals(OBJECT_RESOLVER_REF_TAG)) {
594                         final Element objectResolverRef = (Element) currentNode;
595                         final ObjectResolver subObjectResolver = findObjectResolverByReference(objectResolverRef);
596                         objectResolver.addObjectResolver(subObjectResolver);
597 
598                     } else if (tagName.equals(PARAM_TAG)) {
599                         setParameter(currentElement, objectResolver, connectors);
600                     } else if (tagName.equals(FILTER_TAG)) {
601                         final Element filterElement = (Element) currentNode;
602                         filter = parseFilter(filterElement);
603                     } else {
604                         quietParseUnrecognizedElement(objectResolver, currentElement);
605                     }
606                 }
607             }
608 
609             String isCachedValue = objectResolverElement.getAttribute(CACHED_ATTR);
610             final Boolean isCached = (Boolean) subst(isCachedValue, null, Boolean.class);
611             if (BooleanUtils.isTrue(isCached)) {
612                 if (objectResolver instanceof AbstractObjectResolver) {
613                     ((AbstractObjectResolver) objectResolver).setCached(isCached.booleanValue());
614                 } else {
615                     LOG.warn("Only AbstractObjectResolver can use the attribute cached=\"true\" ");
616                     // TODO hbrabenetz 21.05.2008 : extract setCached into seperate Interface.
617                 }
618             }
619 
620             if (filter != null) {
621                 objectResolver = new FilteredObjectResolverWrapper(objectResolver, filter);
622             }
623         }
624 
625         return objectResolver;
626     }

629     /**
630      * Used internally to parse an mapping element.
631      * 
632      * @param mappingElement The XML Mapping Element.
633      * @return the Map instance (Key = String; Value = String).
634      */
635     protected Map parseMapping(final Element mappingElement) {
636         final String mappingName = mappingElement.getAttribute(NAME_ATTR);
637 
638         Map mapping;
639 
640         String className = mappingElement.getAttribute(CLASS_ATTR);
641         if (StringUtils.isEmpty(className)) {
642             className = "java.util.HashMap";
643         }
644 
645         LOG.debug("Desired Map class: [" + className + ']');
646         try {
647             final Class clazz = loadClass(className);
648             final Constructor constructor = clazz.getConstructor(NO_PARAM);
649             mapping = (Map) constructor.newInstance(null);
650         } catch (final Exception oops) {
651             LOG.error("Could not retrieve mapping [" + mappingName + "]. Reported error follows.", oops);
652             return null;
653         } catch (final NoClassDefFoundError e) {
654             LOG.warn("The Mapping '" + mappingName
655                 + "' cannot be created. There are not all required Libraries inside the Classpath: " + e.getMessage(),
656                 e);
657             return null;
658         }
659 
660         // Setting up a mapping needs to be an atomic operation, in order
661         // to protect potential settings operations while settings
662         // configuration is in progress.
663         synchronized (mapping) {
664 
665             final NodeList children = mappingElement.getChildNodes();
666             final int length = children.getLength();
667 
668             for (int loop = 0; loop < length; loop++) {
669                 final Node currentNode = children.item(loop);
670 
671                 if (currentNode.getNodeType() == Node.ELEMENT_NODE) {
672                     final Element currentElement = (Element) currentNode;
673                     final String tagName = currentElement.getTagName();
674 
675                     if (tagName.equals(ENTRY_TAG)) {
676                         final String key = currentElement.getAttribute(ENTRY_KEY_ATTR);
677                         final String keyRef = currentElement.getAttribute(ENTRY_REFKEY_ATTR);
678                         mapping.put(key, keyRef);
679                     } else {
680                         quietParseUnrecognizedElement(mapping, currentElement);
681                     }
682                 }
683             }
684         }
685 
686         return mapping;
687     }

705         try {
706             final Class clazz = loadClass(className);
707             final Constructor constructor = clazz.getConstructor(NO_PARAM);
708             contentResolver = (ContentResolver) constructor.newInstance(null);
709         } catch (final Exception e) {
710             LOG.error("Could not retrieve contentResolver [" + contentResolverName + "]. " //
711                 + "Reported error follows.", e);
712             return null;
713         } catch (final NoClassDefFoundError e) {
714             LOG.warn("The ContentResolver '" + contentResolverName + "' cannot be created. " //
715                 + "There are not all required Libraries inside the Classpath: " + e.getMessage(), e);
716             return null;
717         }
718 
719         // get connectors for ExpressionLanguage validation
720         final Connector[] connectors = getConnectors(contentResolverElement);
721 
722         Filter filter = null;
723         // Setting up a contentResolver needs to be an atomic operation, in order
724         // to protect potential settings operations while settings
725         // configuration is in progress.
726         synchronized (contentResolver) {
727 
728             final NodeList children = contentResolverElement.getChildNodes();
729             final int length = children.getLength();
730 
731             for (int loop = 0; loop < length; loop++) {
732                 final Node currentNode = children.item(loop);
733 
734                 if (currentNode.getNodeType() == Node.ELEMENT_NODE) {
735                     final Element currentElement = (Element) currentNode;
736                     final String tagName = currentElement.getTagName();
737 
738                     if (tagName.equals(CONTENT_RESOLVER_REF_TAG)) {
739                         final Element contentResolverRef = (Element) currentNode;
740                         final ContentResolver subContentResolver = findContentResolverByReference(contentResolverRef);
741                         contentResolver.addContentResolver(subContentResolver);
742 
743                     } else if (tagName.equals(PARAM_TAG)) {
744                         setParameter(currentElement, contentResolver, connectors);
745                     } else if (tagName.equals(FILTER_TAG)) {
746                         final Element filterElement = (Element) currentNode;
747                         filter = parseFilter(filterElement);
748                     } else {
749                         quietParseUnrecognizedElement(contentResolver, currentElement);
750                     }
751                 }
752             }
753 
754             if (filter != null) {
755                 contentResolver = new FilteredContentResolverWrapper(contentResolver, filter);
756             }
757         }
758 
759         return contentResolver;
760     }

842             final String elementTagName) {
843         // Doesn't work on DOM Level 1 :
844         // Element element = doc.getElementById(nameAttrValue);
845 
846         // Endre's hack:
847         Element element = null;
848         final NodeList list = doc.getElementsByTagName(elementTagName);
849         for (int t = 0; t < list.getLength(); t++) {
850             final Node node = list.item(t);

881     /**
882      * This function will replace expressions like ${connectors.object['']} or simply ${true}.
883      * 
884      * @param value The value or Expression.
885      * @param clazz The expected {@link Class}.
886      * @param connectors required for validating Expression like ${connectors.string['']}
887      * @return the Object-Value of the given value or Expression.
888      */
889     protected Object subst(final String value, final Connector[] connectors, final Class clazz) {
890         if (StringUtils.isEmpty(value)) {
891             return null;
892         }
893 
894         if (value.indexOf("${") >= 0) {
895             try {
896                 final Map context = new HashMap(this.expressionAttributes);
897                 if (connectors != null) {
898                     // Expression like ${connectors.object['...']} or ${connectors.string['...']}
899                     context.put("connectors", new ELConnectorWrapper(connectors));
900 
901                     // Expression like ${connector.FsConnector.object['...']} or
902                     // ${connector.ClassPathConnector.string['...']}
903                     final Map connectorMap = new HashMap();
904                     for (int i = 0; i < connectors.length; i++) {
905                         final Connector connector = connectors[i];
906                         connectorMap.put(connector.getName(), new ELConnectorWrapper(new Connector[] {connector}));
907                     }
908                     context.put("connector", connectorMap);
909                 }
910                 if (SystemUtils.isJavaVersionAtLeast(JAVA_VERSION_1_5)) {
911                     // Expression like ${env['...']} e.g.: ${env['TOMCAT_HOME']} or ${env.TOMCAT_HOME}
912                     // Only since jdk 1.5 ....
913                     context.put("env", System.getenv());
914                 }
915                 final Object result = ExpressionLanguageUtil.evaluateExpressionLanguage(value, context, clazz);
916                 return result;
917             } catch (final Exception e) {
918                 LOG.error(e.getMessage(), e);
919                 return null;
920             }
921         }
922         // else 
923         if (clazz.equals(String.class)) {
924             return value.trim();
925         } else if (clazz.equals(Boolean.class)) {
926             return BooleanUtils.toBooleanObject(value.trim());
927         } else {
928             throw new UnsupportedOperationException("The following Type is not supported now: " + clazz
929                 + "; found value: " + value);
930         }
931     }
