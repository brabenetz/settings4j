17  package org.settings4j.connector;
18  
19  /**
20   * The SystemProperties implementation of an {@link org.settings4j.Connector}.
21   * <p>
22   * 
23   * see also {@link System#getProperty(String, String)}.
24   * 
25   * @author Harald.Brabenetz
26   *
27   */
28  public class SystemPropertyConnector extends AbstractPropertyConnector {
29  
30      /** General Logger for this Class. */
31      private static final org.apache.commons.logging.Log LOG = org.apache.commons.logging.LogFactory
32          .getLog(SystemPropertyConnector.class);
33  
34      /**
35       * Very similar to <code>System.getProperty</code> except that the {@link SecurityException} is hidden.
36       * 
37       * @param key The key to search for.
38       * @param def The default value to return.
39       * @return the string value of the system property, or the default value if there is no property with that key.
40       */
41      protected String getProperty(final String key, final String def) {
42          try {
43              return System.getProperty(key, def);
44          } catch (final Throwable e) { // MS-Java throws com.ms.security.SecurityExceptionEx
45              LOG.debug("Was not allowed to read system property \"" + key + "\".");
46              return def;
47          }
48      }
49  }
