17  package org.settings4j.helper.spring;
18  
19  import org.settings4j.Settings4j;
20  import org.springframework.beans.factory.FactoryBean;
21  import org.springframework.beans.factory.InitializingBean;
22  
23  
24  /**
25   * BeanFactory which gets the Object from {@link Settings4j#getObject(String)}.
26   * <p>
27   * <h2>Simple SpringBean Example:</h2> The following Example can be configured with the settings4j-Key "env/MyVariable".
28   * <br />
29   * e.g.: System.setProperty("env/MyVariable", "Hallo World");<br />
30   * OR with JNDI-Context OR in a Classpath-File under "classpath:env/MyVariable" Or with a custom
31   * {@link org.settings4j.Connector}-Implementation.
32   * 
33   * <pre>
34   *  &lt;bean id="MyConfigurableValue" class="org.settings4j.helper.spring.Settings4jFactoryBean"&gt;
35   *      &lt;property name="key"&gt;&lt;value&gt;<b>env/MyVariable</b>&lt;/value&gt;&lt;/property&gt;
36   *  &lt;/bean&gt;
37   * </pre>
38   * 
39   * <h2>More complex SpringBean Example:</h2> This Example shows how a Hibernate SessionFactory can optional customized
40   * with additional HibernateProperties.
41   * 
42   * <pre>
43   *  &lt;bean id="hibernateProperties"
44   *      class="org.springframework.beans.factory.config.PropertiesFactoryBean"&gt;
45   *      &lt;property name="locations"&gt;
46   *          &lt;list&gt;
47   *              &lt;value&gt;classpath:org/settings4j/helper/spring/DefaultHibernate.properties&lt;/value&gt;
48   *              &lt;bean class="org.settings4j.helper.spring.Settings4jFactoryBean"&gt;
49   *                  &lt;property name="key"&gt;
50   *                      &lt;value&gt;<b>env/MyCustomProprtiesLocationKey</b>&lt;/value&gt;
51   *                  &lt;/property&gt;
52   *                  &lt;property name="defaultObject"&gt;
53   *                      &lt;!-- This could also be an empty Property-File But it must Exist. --&gt;
54   *                      &lt;!-- PropertiesFactoryBean cannot handle invalid Paths. --&gt;
55   *                      &lt;value&gt;classpath:org/settings4j/helper/spring/DefaultHibernate.properties&lt;/value&gt;
56   *                  &lt;/property&gt;
57   *              &lt;/bean&gt;
58   *          &lt;/list&gt;
59   *      &lt;/property&gt;
60   *  &lt;/bean&gt;
61   *  
62   *  &lt;bean id="mySessionFactory" class="org.springframework.orm.hibernate3.LocalSessionFactoryBean"&gt;
63   *      &lt;property name="dataSource" ref="db2Datasource2" /&gt;
64   *      &lt;property name="hibernateProperties" ref="hibernateProperties" /&gt;
65   *      ....
66   *  &lt;/bean&gt;
67   * </pre>
68   * 
69   * @author Harald.Brabenetz
70   */
71  public class Settings4jFactoryBean implements FactoryBean, InitializingBean {
72  
73      private String key;
74      private boolean singleton = true;
75      private Object singletonObject;
76      private Object defaultObject;
77      private Class expectedType;
78  
79      public String getKey() {
80          return this.key;
81      }
82  
83      /**
84       * @param key - the Setting4j-Key for Your Configuration-Value.
85       */
86      public void setKey(final String key) {
87          this.key = key;
88      }
89  
90      /**
91       * @param singleton - If <code>true</code>, the Value will be read during SpringBean-Initialization. Default is
92       *            <code>true</code>
93       */
94      public void setSingleton(final boolean singleton) {
95          this.singleton = singleton;
96      }
97  
98      /**
99       * Specify the type that the configured object is supposed to be assignable to, if any.
100      * 
101      * @param expectedType the expected {@link Class}.
102      */
103     public void setExpectedType(final Class expectedType) {
104         this.expectedType = expectedType;
105     }
106 
107     /**
108      * Return the type that the configured object is supposed to be assignable to, if any.
109      * 
110      * @return the type that the configured object is supposed to be assignable to.
111      */
112     public Class getExpectedType() {
113         return this.expectedType;
114     }
115 
116 
117     public Object getDefaultObject() {
118         return this.defaultObject;
119     }
120 
121 
122     public void setDefaultObject(final Object defaultObject) {
123         this.defaultObject = defaultObject;
124     }
125 
126     /** {@inheritDoc} */
127     public Object getObject() {
128         if (this.singleton) {
129             return this.singletonObject;
130         }
131         return getSettings4jObject();
132     }
133 
134     /** {@inheritDoc} */
135     public Class getObjectType() {
136         final Object obj = getObject();
137         if (obj != null) {
138             return obj.getClass();
139         }
140         return getExpectedType();
141     }
142 
143     /** {@inheritDoc} */
144     public boolean isSingleton() {
145         return this.singleton;
146     }
147 
148     /** {@inheritDoc} */
149     public void afterPropertiesSet() throws Exception {
150         if (this.singleton) {
151             this.singletonObject = getSettings4jObject();
152         }
153     }
154 
155     private Object getSettings4jObject() {
156         Object result = Settings4j.getObject(this.key);
157         if (result == null) {
158             result = Settings4j.getString(this.key);
159         }
160 
161         if (result == null) {
162             result = this.defaultObject;
163         }
164         return result;
165     }

167 }
