119     public void testJNDIConnectorWithJNDI() throws Exception {
120         LOG.info("START: testJNDIConnectorWithJNDI");
121         // Set JNDI Tomcat Configs
122         setTomcatJNDIContextProperties();
123 
124         final String testTextPath = this.testDir.getAbsolutePath() + "/test.txt";
125 
126         JNDIConnector connector;
127         int saveStatus;
128         String resultString;
129         byte[] resultContent;
130         Object resultObject;
131 
132         connector = new JNDIConnector();
133 
134         resultString = connector.getString("helloWorldPath");
135         assertNull(resultString);
136 
137         resultContent = connector.getContent("helloWorldPath");
138         assertNull(resultContent);
139 
140         resultObject = connector.getObject("helloWorldPath");
141         assertNull(resultObject);
142 
143         // set the PATH into the JNDI-Context
144         saveStatus = connector.setObject("helloWorldPath", testTextPath);
145         assertEquals(Constants.SETTING_SUCCESS, saveStatus);
146         resultString = connector.getString("helloWorldPath");
147         assertEquals(testTextPath, resultString);
148 
149         // if no ContentResolver is available. the value will be stored directly into the JNDI-Context
150         saveStatus = connector.setObject("helloWorldPath", "Hello World".getBytes(this.charset));
151         assertEquals(Constants.SETTING_SUCCESS, saveStatus);
152         // The String-Value cannot read a byte[]
153         resultString = connector.getString("helloWorldPath");
154         assertNull(resultString);
155         // The Content-Value should be the same
156         resultContent = connector.getContent("helloWorldPath");
157         assertNotNull(resultContent);
158         assertEquals("Hello World", new String(resultContent, this.charset));
159 
160 
161         // No Exception if JNDIContext not available
162         saveStatus = connector.setObject("helloWorldPath", testTextPath);
163         assertEquals(Constants.SETTING_SUCCESS, saveStatus);
164         resultString = connector.getString("helloWorldPath");
165         assertEquals(testTextPath, resultString);
166         resultObject = connector.getObject("helloWorldPath");
167         assertEquals(testTextPath, resultObject);
168 
169         // if no ObjectResolver is available. the value will be stored directly into the JNDI-Context
170         final ContentResolver contentResolver = new UnionContentResolver();
171         final FSContentResolver fsContentResolver = new FSContentResolver();
172         fsContentResolver.setRootFolderPath(this.testDir.getAbsolutePath()); // should also work with temp-folder
173         contentResolver.addContentResolver(fsContentResolver);
174         contentResolver.addContentResolver(new ClasspathContentResolver());
175         connector.setContentResolver(contentResolver);
176 
177 
178         // set the PATH into the JNDI-Context
179         saveStatus = connector.setObject("helloWorldPath", testTextPath);
180         assertEquals(Constants.SETTING_SUCCESS, saveStatus);
181         resultString = connector.getString("helloWorldPath");
182         assertEquals(testTextPath, resultString);
183 
184         // Store the content into the file on the file system.
185         // And save the Path in the JNDI Context.
186         FileUtils.writeStringToFile(new File(testTextPath), "Hello World FileContent");
187         saveStatus = connector.setObject("helloWorldPath", testTextPath);
188 
189         assertEquals(Constants.SETTING_SUCCESS, saveStatus);
190         // The String-Value should be the text-File Path.
191         resultString = connector.getString("helloWorldPath");
192         assertEquals(testTextPath, resultString);
193         // The Content-Value should be content of the file.
194         resultContent = connector.getContent("helloWorldPath");
195         assertNotNull(resultContent);
196         assertEquals("Hello World FileContent", new String(resultContent, this.charset));
197 
198 
199     }

201     public void testJNDIConnectorIsAvailable() throws Exception {
202         LOG.info("START: testJNDIConnectorIsAvailable");
203         JNDIConnector connector;
204 
205         connector = new JNDIConnector();
206 
207         // JNDIContext not available
208         assertNull(connector.getIsJNDIAvailable());
209         assertFalse(connector.isJNDIAvailable()); // will set the boolean Flag
210         assertNotNull(connector.getIsJNDIAvailable());
211         assertFalse(connector.isJNDIAvailable());
212 
213         // Set JNDI Tomcat Configs (Enable JNDI Context)
214         setTomcatJNDIContextProperties();
215 
216         // JNDI-Connector is disabled:
217         connector.setObject("irgendwas", "Something to create parent Context");
218         assertNull(connector.getObject("irgendwas"));
219 
220         // reset boolean Flag
221         connector.setIsJNDIAvailable(null);
222 
223         // JNDIContext is now available
224         assertNull(connector.getIsJNDIAvailable());
225         assertTrue(connector.isJNDIAvailable()); // will set the boolean Flag again
226         assertNotNull(connector.getIsJNDIAvailable());
227         assertTrue(connector.isJNDIAvailable());
228 
229 
230         // test setObject and getObject.
231         assertNull(connector.getObject("irgendwas"));
232         connector.setObject("irgendwas", "Something to test");
233         assertEquals("Something to test", connector.getObject("irgendwas"));
234 
235     }

237     public static void setTomcatJNDIContextProperties() {
238         System.setProperty(Context.INITIAL_CONTEXT_FACTORY, "org.apache.naming.java.javaURLContextFactory");
239         System.setProperty(Context.PROVIDER_URL, "localhost:1099");
240         System.setProperty(Context.URL_PKG_PREFIXES, "org.apache.naming");
241 
242     }

244     private void removeJNDIContextProperties() {
245         System.getProperties().remove(Context.INITIAL_CONTEXT_FACTORY);
246         System.getProperties().remove(Context.PROVIDER_URL);
247         System.getProperties().remove(Context.URL_PKG_PREFIXES);
248     }
