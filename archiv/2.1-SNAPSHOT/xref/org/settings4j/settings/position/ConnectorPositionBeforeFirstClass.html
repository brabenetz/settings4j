20  package org.settings4j.settings.position;
21  
22  import java.util.List;
23  
24  import org.settings4j.Connector;
25  import org.settings4j.ConnectorPosition;
26  
27  /**
28   * Iterates thought the existing connectors and return the position before the first occurrence of the given Connector-class.
29   *
30   * @author brabenetz
31   */
32  public class ConnectorPositionBeforeFirstClass implements ConnectorPosition {
33  
34      private final Class<? extends Connector> connectorClass;
35  
36      /**
37       * @param connectorClass
38       *        the connector class to search in the given connectors list.
39       */
40      public ConnectorPositionBeforeFirstClass(final Class<? extends Connector> connectorClass) {
41          super();
42          this.connectorClass = connectorClass;
43      }
44  
45      @Override
46      public int getPosition(final List<Connector> connectors) {
47          int pos = 0;
48          for (final Connector connector : connectors) {
49              if (this.connectorClass.equals(connector.getClass())) {
50                  return pos;
51              }
52              pos++;
53          }
54          return UNKNOWN_POSITION;
55      }
56  
57  }
