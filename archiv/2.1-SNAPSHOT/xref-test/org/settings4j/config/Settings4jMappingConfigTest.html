20  package org.settings4j.config;
21  
22  import static org.hamcrest.MatcherAssert.assertThat;
23  import static org.hamcrest.Matchers.is;
24  
25  import org.junit.Test;
26  import org.settings4j.Settings4jInstance;
27  import org.settings4j.Settings4jRepository;
28  import org.settings4j.test.TestUtils;
29  
30  public class Settings4jMappingConfigTest extends AbstractTestSettings4jConfig {
31  
32      /** General Logger for this Class. */
33      private static final org.slf4j.Logger LOG = org.slf4j.LoggerFactory.getLogger(Settings4jMappingConfigTest.class);
34  
35      @Test
36      public void testMapping() {
37          LOG.debug("run TestSettings4jMappingConfig.testMapping");
38          final Settings4jRepository settingsRepository = TestUtils
39              .getConfiguredSettingsRepository("org/settings4j/config/testConfigMapping.xml");
40  
41          final Settings4jInstance mycompanySeetings = settingsRepository.getSettings();
42  
43          final String webservice = "http://settings4j.org/XY";
44          System.setProperty("global/webserviceXY", webservice);
45  
46          final String moduleXWS = mycompanySeetings.getString("com/mycompany/moduleX/webserviceXY");
47          final String moduleYWS = mycompanySeetings.getString("com/mycompany/moduleY/webserviceXY");
48  
49          assertThat(moduleXWS, is(webservice));
50          assertThat(moduleYWS, is(webservice));
51  
52      }
53  }
