36  package org.settings4j.connector;
37  
38  import java.util.prefs.BackingStoreException;
39  import java.util.prefs.Preferences;
40  
41  import org.settings4j.Settings4j;
42  
43  import junit.framework.TestCase;
44  
45  /**
46   * Test Suite for {@link PreferencesConnector}. Checkstyle:OFF .*
47   *
48   * @author brabenetz
49   */
50  public class PreferencesConnectorTest extends TestCase {
51  
52  
53      private static final String PREF_UNITTEST_NODE = "org/settings4j/unittest";
54  
55      @Override
56      protected void setUp() throws Exception {
57          removeUnitTestNode(Preferences.userRoot());
58          removeUnitTestNode(Preferences.systemRoot());
59      }
60  
61      /**
62       * Smoke Test if the Preferences from JDK works as expected.
63       * <p>
64       * This Test tests the PRE-REQUIREMENTS for the PreferencesConnector-tests and NOT the {@link PreferencesConnector} itself.
65       * </p>
66       *
67       * @throws BackingStoreException
68       *         in case of an error.
69       */
70      public void testSmokeTest() throws BackingStoreException {
71  
72          final Preferences myModulePrefs = Preferences.userRoot().node("com/myapp/mymodule");
73          // put test value
74          myModulePrefs.put("myKey", "myNewValue");
75          // read TestValue
76          final String myValue = myModulePrefs.get("myKey", null);
77  
78          // validate Result
79          assertEquals("myNewValue", myValue);
80  
81          // remove node
82          myModulePrefs.removeNode();
83          assertEquals(false, Preferences.userRoot().nodeExists("com/myapp/mymodule"));
84      }
85  
86      private void removeUnitTestNode(final Preferences userRoot) throws BackingStoreException {
87          if (userRoot.nodeExists(PREF_UNITTEST_NODE)) {
88              userRoot.node(PREF_UNITTEST_NODE).removeNode();
89          }
90      }
91  
92      public void testSaveSuccessSlash() {
93          final PreferencesConnector connector = new PreferencesConnector();
94          connector.setString(PREF_UNITTEST_NODE + "/testSaveSuccessSlash", "test");
95  
96          assertEquals("test", Preferences.userRoot().node(PREF_UNITTEST_NODE).get("testSaveSuccessSlash", null));
97      }
98  
99      public void testSaveSuccessBackslash() {
100         final PreferencesConnector connector = new PreferencesConnector();
101         connector.setString(PREF_UNITTEST_NODE + "\\testSaveSuccessBackslash", "test");
102 
103         assertEquals("test", Preferences.userRoot().node(PREF_UNITTEST_NODE).get("testSaveSuccessBackslash", null));
104     }
105 
106     public void testReadSuccess() {
107         Preferences.userRoot().node(PREF_UNITTEST_NODE).put("testReadSuccess", "test");
108 
109         final PreferencesConnector connector = new PreferencesConnector();
110 
111         assertEquals("test", connector.getString(PREF_UNITTEST_NODE + "/testReadSuccess"));
112     }
113 
114     public void testReadFail() {
115         final PreferencesConnector connector = new PreferencesConnector();
116 
117         assertNull(connector.getString(PREF_UNITTEST_NODE + "/testReadFailDoesntExist"));
118         assertNull(connector.getString("testReadFailDoesntExist"));
119     }
120 
121     public void testRoundup() {
122         final PreferencesConnector connector = new PreferencesConnector();
123 
124         assertNull(connector.getString(PREF_UNITTEST_NODE + "/testRoundup"));
125         // 1. save
126         connector.setString(PREF_UNITTEST_NODE + "/testRoundup", "test");
127         // 2. read
128         assertEquals("test", connector.getString(PREF_UNITTEST_NODE + "/testRoundup"));
129     }
130 
131     public void testFormularExample() {
132         final PreferencesConnector connector = new PreferencesConnector();
133         connector.setString("com/mycompany/mycalculation/my-formula", "a * a - 2");
134 
135         assertEquals("a * a - 2", Settings4j.getString("com/mycompany/mycalculation/my-formula"));
136     }
137 }
