20  package org.settings4j.connector;
21  
22  import static org.hamcrest.MatcherAssert.assertThat;
23  import static org.hamcrest.Matchers.containsString;
24  import static org.hamcrest.Matchers.is;
25  
26  import java.io.File;
27  import java.io.IOException;
28  import java.net.URL;
29  
30  import org.junit.Rule;
31  import org.junit.Test;
32  import org.junit.rules.ExpectedException;
33  import org.junit.runner.RunWith;
34  import org.mockito.runners.MockitoJUnitRunner;
35  
36  @RunWith(MockitoJUnitRunner.class)
37  public class PropertyFileConnectorTest {
38  
39      @Rule
40      public ExpectedException thrown = ExpectedException.none();
41  
42      @Test
43      public void testReadPropertyFromFilePath() throws IOException {
44          final PropertyFileConnector connector = new PropertyFileConnector();
45          connector.setPropertyFromPath("file:./src/test/resources/org/settings4j/connector/propertyFile.properties");
46          connector.setResolveRelativePaths(true);
47          assertThat(connector.getString("xyz"), is("Value from Property-File"));
48          assertThat(connector.getString("someRelativePath"),
49              is(new File("./src/test/resources/org/settings4j/connector/test.xml").getCanonicalFile().toURI().toURL().toExternalForm()));
50          assertThat(connector.getString("someRelativeSiblingPath"),
51              is(new File("./src/test/resources/org/settings4j/test1/test.xml").getCanonicalFile().toURI().toURL().toExternalForm()));
52          assertThat(connector.getString("someNormalWindowsPath"), is(new URL("file:C:/test.xml").toExternalForm()));
53          assertThat(connector.getString("someNormalUnixPath"), is(new URL("file:/test.xml").toExternalForm()));
54      }
55  
56      @Test
57      public void testReadPropertyFromClasspath() throws IOException {
58          final PropertyFileConnector connector = new PropertyFileConnector();
59          connector.setResolveRelativePaths(true);
60          connector.setPropertyFromPath("classpath:/org/settings4j/connector/propertyFile.properties");
61          assertThat(connector.getString("xyz"), is("Value from Property-File"));
62          assertThat(connector.getString("someRelativePath"),
63              is(new File("./target/test-classes/org/settings4j/connector/test.xml").getCanonicalFile().toURI().toURL().toExternalForm()));
64          assertThat(connector.getString("someRelativeSiblingPath"),
65              is(new File("./target/test-classes/org/settings4j/test1/test.xml").getCanonicalFile().toURI().toURL().toExternalForm()));
66          assertThat(connector.getString("someNormalWindowsPath"), is(new URL("file:C:/test.xml").toExternalForm()));
67          assertThat(connector.getString("someNormalUnixPath"), is(new URL("file:/test.xml").toExternalForm()));
68      }
69  
70      @Test
71      public void testReadPropertyFromJarFile() {
72          final PropertyFileConnector connector = new PropertyFileConnector();
73          connector.setResolveRelativePaths(true);
74          connector.setPropertyFromPath("classpath:/META-INF/maven/org.slf4j/slf4j-api/pom.properties");
75          assertThat(connector.getString("groupId"), is("org.slf4j"));
76          assertThat(connector.getPropertyFileFolderUrl().toExternalForm() + "pom.properties",
77              is(Thread.currentThread().getContextClassLoader().getResource("META-INF/maven/org.slf4j/slf4j-api/pom.properties").toExternalForm()));
78      }
79  
80      @Test
81      public void testReadPropertyFromClasspathWhichDoesnotExist() {
82          this.thrown.expect(NullPointerException.class);
83          this.thrown.expectMessage(containsString("classpath:/NotExisting.properties"));
84  
85          final PropertyFileConnector connector = new PropertyFileConnector();
86          connector.setResolveRelativePaths(true);
87          connector.setPropertyFromPath("classpath:/NotExisting.properties");
88      }
89  
90  }
