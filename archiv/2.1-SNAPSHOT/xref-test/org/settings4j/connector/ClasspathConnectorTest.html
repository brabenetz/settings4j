20  package org.settings4j.connector;
21  
22  import static org.hamcrest.MatcherAssert.assertThat;
23  import static org.hamcrest.Matchers.containsString;
24  import static org.hamcrest.Matchers.is;
25  import static org.hamcrest.Matchers.nullValue;
26  
27  import java.io.File;
28  import java.nio.charset.UnsupportedCharsetException;
29  
30  import org.apache.commons.io.FileUtils;
31  import org.junit.After;
32  import org.junit.Before;
33  import org.junit.Rule;
34  import org.junit.Test;
35  import org.junit.rules.ExpectedException;
36  import org.junit.runner.RunWith;
37  import org.mockito.InjectMocks;
38  import org.mockito.Matchers;
39  import org.mockito.Mock;
40  import org.mockito.Mockito;
41  import org.mockito.runners.MockitoJUnitRunner;
42  import org.settings4j.ContentResolver;
43  import org.settings4j.ObjectResolver;
44  import org.settings4j.contentresolver.ClasspathContentResolver;
45  
46  @RunWith(MockitoJUnitRunner.class)
47  public class ClasspathConnectorTest {
48  
49      private static final String TEST_VALUE_STRING = "Test-Value";
50  
51      private static final byte[] TEST_VALUE_CONTENT = "Test-Value".getBytes();
52  
53      private static final Object TEST_VALUE_OBJECT = "Test-Value";
54  
55      private static final String TEST_KEY = "TEST";
56  
57      @InjectMocks
58      private ClasspathConnector cpConnector;
59  
60      @Mock
61      private ClasspathContentResolver classpathContentResolver;
62  
63      private File testDir;
64  
65      @Rule
66      public ExpectedException thrown = ExpectedException.none();
67  
68      @Before
69      public void setUp() throws Exception {
70          this.testDir = new File("target/test/" + getClass().getSimpleName() + "/".toLowerCase()).getAbsoluteFile();
71      }
72  
73      @After
74      public void tearDown() throws Exception {
75          FileUtils.deleteDirectory(this.testDir);
76      }
77  
78      @Test
79      public void testGetContent() throws Exception {
80          // first Test
81          assertThat(this.cpConnector.getContent(TEST_KEY), is(nullValue()));
82  
83          // second Test
84          Mockito.when(this.classpathContentResolver.getContent(TEST_KEY)).thenReturn(TEST_VALUE_CONTENT);
85          assertThat(this.cpConnector.getContent(TEST_KEY), is(TEST_VALUE_CONTENT));
86  
87      }
88  
89      @Test
90      public void testGetObject() throws Exception {
91          // first Test without ObjectResolver
92          assertThat(this.cpConnector.getObject(TEST_KEY), is(nullValue()));
93  
94          // second Test with Object Resolver but returns null
95          final ObjectResolver objResolver = Mockito.mock(ObjectResolver.class);
96          this.cpConnector.setObjectResolver(objResolver);
97          assertThat(this.cpConnector.getObject(TEST_KEY), is(nullValue()));
98  
99          // third Test
100         Mockito.when(objResolver.getObject(Matchers.eq(TEST_KEY), Matchers.any(ContentResolver.class))).thenReturn(TEST_VALUE_OBJECT);
101         assertThat(this.cpConnector.getObject(TEST_KEY), is(TEST_VALUE_OBJECT));
102 
103     }
104 
105     @Test
106     public void testGetString() throws Exception {
107         // first Test without ObjectResolver
108         assertThat(this.cpConnector.getString(TEST_KEY), is(nullValue()));
109 
110         // second Test with Object Resolver but returns null
111         Mockito.when(this.classpathContentResolver.getContent(TEST_KEY)).thenReturn(TEST_VALUE_CONTENT);
112         assertThat(this.cpConnector.getString(TEST_KEY), is(TEST_VALUE_STRING));
113 
114         // third Test
115         assertThat(this.cpConnector.getString(TEST_KEY), is(TEST_VALUE_STRING));
116 
117     }
118 
119     @Test
120     public void testSetContentResolver() throws Exception {
121         final ContentResolver contentResolver = new ClasspathContentResolver();
122         assertThat(this.cpConnector.getContentResolver(), is(nullValue()));
123 
124         // run Test
125         this.cpConnector.setContentResolver(contentResolver);
126 
127         // validate result
128         assertThat(this.cpConnector.getContentResolver(), is(contentResolver));
129         assertThat(this.cpConnector.getContentResolver() == contentResolver, is(true));
130 
131     }
132 
133     @Test
134     public void testSetCharset() throws Exception {
135         // run test
136         this.cpConnector.setCharset("ISO-8859-1");
137 
138         // validate
139         assertThat(this.cpConnector.getCharset(), is("ISO-8859-1"));
140 
141     }
142 
143     @Test
144     public void testSetCharsetShouldThrowIllegalCharsetNameException() throws Exception {
145         this.thrown.expect(UnsupportedCharsetException.class);
146         this.thrown.expectMessage(containsString("xyz"));
147         // run test
148         this.cpConnector.setCharset("xyz");
149     }
150 }
