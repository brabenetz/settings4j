276     private void initLog4jConfigurationHappyPath(final String log4jConfigXml) {
277         final MockServletContext servletContext = prepareServletContext(//
278             log4jConfigXml, //
279             LOG4J_CONFIG_KEY);
280 
281         // start test: initLog4jConfiguration from ServletContext initParameter
282         final Log4jConfigurationLoader log4jConfigurationLoader = new Log4jConfigurationLoader();
283         log4jConfigurationLoader.initLog4jConfiguration(servletContext);
284 
285         // assert PreReQuirements
286         InMemoryLog4jAppender.linesClear();
287     }
288 
289     private MockServletContext prepareServletContext(final String log4DefaultConfiguration,
290         final String log4ConfigurationKey) {
291         final StringBuffer propertiesString = new StringBuffer();
292         if (StringUtils.isNotEmpty(log4DefaultConfiguration)) {
293             propertiesString.append(log4ConfigurationKey);
294             propertiesString.append("=");
295             propertiesString.append(log4DefaultConfiguration);
296         }
297 
298         final MockServletContext servletContext = new MockServletContext();
299         servletContext.addInitParameter(DefaultPropertiesLoader.DEFAULT_PROPERTIES, propertiesString.toString());
300         servletContext.addInitParameter(Log4jConfigurationLoader.LOG4J_CONFIG_SETTINGS4JKEY, log4ConfigurationKey);
301         return servletContext;
302     }
