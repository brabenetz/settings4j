20  package org.settings4j.config;
21  
22  import static org.hamcrest.MatcherAssert.assertThat;
23  import static org.hamcrest.Matchers.is;
24  import static org.hamcrest.Matchers.nullValue;
25  
26  import org.junit.After;
27  import org.junit.Before;
28  import org.junit.Test;
29  import org.settings4j.Settings4j;
30  import org.settings4j.test.TestUtils;
31  
32  public class DefaultFallbackConfigurationTest {
33  
34      /** General Logger for this Class. */
35      private static final org.slf4j.Logger LOG = org.slf4j.LoggerFactory
36          .getLogger(DefaultFallbackConfigurationTest.class);
37  
38      @Before
39      public void setUp() throws Exception {
40          TestUtils.reconfigureSettings4jWithDefaultConfig();
41          System.setProperty("testDefaultProperty", "HelloWorld");
42      }
43  
44      @After
45      public void tearDown() throws Exception {
46          System.clearProperty("testDefaultProperty");
47      }
48  
49      /**
50       * Test parsing of defaultsettings4j.xml (FALLBACK-Configuration).
51       */
52      @Test
53      public void testDefaultSettings4jConfig() {
54  
55          LOG.debug(//
56              "#### AFTER THE FOLLOWING LINE SETTINGS4J WILL BE " //
57              + "CONFIGURED THE FIRST TIME WITH THE FALLBACK-FILE #####");
58          assertThat(Settings4j.getString("testDefaultProperty"), is("HelloWorld"));
59          LOG.debug("#### FINISH CONFIGURATION #####");
60  
61          // check if there is no Exception thrown:
62          assertThat(Settings4j.getString("xyz"), is(nullValue()));
63          assertThat(Settings4j.getContent("xyz"), is(nullValue()));
64          assertThat(Settings4j.getObject("xyz"), is(nullValue()));
65  
66          // the Default settings Configuration is readonly
67          final int expectedConnectors = 5;
68          assertThat(Settings4j.getConnectors().size(), is(expectedConnectors));
69      }
70  }
