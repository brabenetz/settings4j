37       * Test parsing of defaultsettings4j.xml (FALLBACK-Configuration)

39      public void testDefaultSettings4jConfig(){
40          SettingsRepository settingsRepository = UtilTesting.getConfiguredSettingsRepository(SettingsManager.DEFAULT_FALLBACK_CONFIGURATION_FILE);
41          
42          HierarchicalSettings rootSeetings = (HierarchicalSettings)settingsRepository.getRootSettings();
43          
44          // rootSettings are extra. So the default Settings have no settings
45          assertEquals(0, settingsRepository.getCurrentSettingsList().size());
46  
47          // create a Custom Settings-Object for unittest
48          // normal usecase:
49          // Settings SETTINGS = Settings.getSettings(MyClass.class);
50          Settings mycompanySeetings = settingsRepository.getSettings("com.mycompany");
51  
52          // now we have a Settings inside the settingsRepository
53          assertEquals(1, settingsRepository.getCurrentSettingsList().size());
54          
55          
56          //the rootSettings have three Connectors
57          assertEquals(3, rootSeetings.getConnectors().size());
58  
59          //check if there is no Exception thrown:
60          assertNull(rootSeetings.getString("xyz"));
61          assertNull(rootSeetings.getContent("xyz"));
62          assertNull(rootSeetings.getObject("xyz"));
63          assertNull(mycompanySeetings.getString("xyz"));
64          assertNull(mycompanySeetings.getContent("xyz"));
65          assertNull(mycompanySeetings.getObject("xyz"));
66  
67          //check if there is a Exception thrown:
68          try {
69              rootSeetings.setString("xyz", "xyz");
70              fail("must throw an NoWriteableConnectorFoundException");
71          } catch (NoWriteableConnectorFoundException e) {
72              assertEquals("Content 'xyz' cannot be writen. No writeable Connector found", e.getMessage());
73          }
74  
75          try {
76              rootSeetings.setContent("xyz", "xyz".getBytes());
77              fail("must throw an NoWriteableConnectorFoundException");
78          } catch (NoWriteableConnectorFoundException e) {
79              assertEquals("Content 'xyz' cannot be writen. No writeable Connector found", e.getMessage());
80          }
81  
82          try {
83              rootSeetings.setObject("xyz", "xyz");
84              fail("must throw an NoWriteableConnectorFoundException");
85          } catch (NoWriteableConnectorFoundException e) {
86              assertEquals("Content 'xyz' cannot be writen. No writeable Connector found", e.getMessage());
87          }
88          
89          assertEquals(3, settingsRepository.getConnectorCount());
90      }

99      public void testFSConfigTempFolder(){
100         SettingsRepository settingsRepository = UtilTesting.getConfiguredSettingsRepository("org/settings4j/config/testConfigFSTempfolder.xml");
101 
102         Settings mycompanySeetings = settingsRepository.getSettings("com.mycompany");
103 
104         // check if there is a Exception thrown:
105         // only com.mycompany.myapp and above can write
106         try {
107             mycompanySeetings.setString("xyz", "abc");
108             fail("must throw an NoWriteableConnectorFoundException");
109         } catch (NoWriteableConnectorFoundException e) {
110             assertEquals(NoWriteableConnectorFoundException.NO_WRITEABLE_CONNECTOR_FOUND_1, e.getKey());
111             assertEquals("Content 'xyz' cannot be writen. No writeable Connector found", e.getMessage());
112         }
113         
114         // store values into the default java temporary directory with subfolder "Settings4j"
115         // String tmpdir = System.getProperty("java.io.tmpdir");
116         Settings settings1 = settingsRepository.getSettings("com.mycompany.myapp");
117         File tmpFolder = UtilTesting.getTmpFolder();
118         File fileXyz = new File(tmpFolder, "xyz");
119         assertFalse(fileXyz.exists());
120         settings1.setString("xyz", "abc");
121         assertTrue(fileXyz.exists());
122 
123         Settings settings2 = settingsRepository.getSettings("com.mycompany.myapp.subcomponent");
124         settings2.setString("xyz2", "abc2");
125         
126         //every settings have read access to the same FSConnector.
127         assertEquals("abc", settings2.getString("xyz"));
128         assertEquals("abc2", settings1.getString("xyz2"));
129         assertEquals("abc2", mycompanySeetings.getString("xyz2"));
130 
131         assertEquals(1, settingsRepository.getConnectorCount());
132     }

207      */
208     public void testObjectResolverConfig4Spring() throws Exception{
209         
210         SettingsRepository settingsRepository = UtilTesting.getConfiguredSettingsRepository("org/settings4j/config/testConfigObjectResolver4Spring.xml");
211 
212         String key1 = "org/settings4j/objectResolver/testSpring1"; // spring-configuration
213         String key2 = "org/settings4j/objectResolver/test1"; // XML-Bean-Configuration
214 
215         // store values into the default java temporary directory with subfolder "Settings4j"
216         // String tmpdir = System.getProperty("java.io.tmpdir");
217         Settings settings1 = settingsRepository.getSettings("com.mycompany.myapp");
218         
219         // the propety-file "org/settings4j/objectResolver/test1.properties must exists"
220         DataSource dataSource = (DataSource)settings1.getObject(key1);
221         assertNotNull(dataSource);
222         
223         settings1.setObject(key2, dataSource);
224 
225         DataSource dataSource2 = (DataSource)settings1.getObject(key2);
226         assertNotNull(dataSource2);
227         assertTrue(dataSource != dataSource2);
228 
229         
230         // test DataSource
231         Connection conn = null;
232         try{
233             conn = dataSource2.getConnection();
234             PreparedStatement pstmt = conn.prepareStatement("create Table test ( name VARCHAR )");
235             pstmt.execute();
236             pstmt.close();
237 
238             pstmt = conn.prepareStatement("insert into test(name) values (?)");
239             pstmt.setString(1, "Hello World");
240             pstmt.execute();
241             pstmt.close();
242             
243             pstmt = conn.prepareStatement("select * from test");
244             ResultSet rs = pstmt.executeQuery();
245             rs.next();
246             String result = rs.getString(1);
247             rs.close();
248             pstmt.close();
249 
250             pstmt = conn.prepareStatement("drop Table test");
251             pstmt.execute();
252             pstmt.close();
253             
254             assertEquals("Hello World", result);
255             
256         } finally {
257             if (conn != null){
258                 conn.close();
259             }
260         }
261     }
