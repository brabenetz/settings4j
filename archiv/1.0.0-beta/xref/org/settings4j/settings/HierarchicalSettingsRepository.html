32  /**
33   * This class is specialized in retrieving settings by name and also maintaining the settings
34   * hierarchy.
35   * 
36   * <p>
37   * <em>The casual user does not have to deal with this class
38   * directly.</em>
39   * 
40   * <p>
41   * The structure of the settings hierarchy is maintained by the {@link #getSettings} method. The
42   * hierarchy is such that children link to their parent but parents do not have any pointers to
43   * their children. Moreover, settings can be instantiated in any order, in particular descendant
44   * before ancestor.
45   * 
46   * <p>
47   * In case a descendant is created before a particular ancestor, then it creates a provision node
48   * for the ancestor and adds itself to the provision node. Other descendants of the same ancestor
49   * add themselves to the previously created provision node.
50   * 
51   * 
52   * @author hbrabenetz
53   * @author Ceki G&uuml;lc&uuml; (the original log4j org.apache.log4j.Hierarchy)
54   * 
55   */

92      public List getCurrentSettingsList() {
93          // The accumlation in list is necessary because not all elements in
94          // ht are HierarchicalSettings objects as there might be some ProvisionNodes
95          // as well.
96          List list = new ArrayList(settingsMap.size());
97  
98          Iterator iterator = settingsMap.values().iterator();
99          while (iterator.hasNext()) {
100             Object o = iterator.next();
101             if (o instanceof HierarchicalSettings) {

103             }
104         }

116     public Settings getSettings(String name, SettingsFactory factory) {
117         // System.out.println("getInstance("+name+") called.");
118         SettingsKey key = new SettingsKey(name);
119         // Synchronize to prevent write conflicts. Read conflicts (in
120         // getChainedLevel method) are possible only if variable
121         // assignments are non-atomic.
122         HierarchicalSettings settings;
123 
124         synchronized (settingsMap) {
125             Object o = settingsMap.get(key);
126             if (o == null) {
127                 settings = (HierarchicalSettings) factory.makeNewSettingsInstance(name);
128                 settings.setSettingsRepository(this);
129                 settingsMap.put(key, settings);
130                 updateParents(settings);
131                 return settings;
132             } else if (o instanceof HierarchicalSettings) {
133                 return (HierarchicalSettings) o;
134             } else if (o instanceof ProvisionNode) {
135                 // System.out.println("("+name+") ht.get(this) returned ProvisionNode");
136                 settings = (HierarchicalSettings) factory.makeNewSettingsInstance(name);
137                 settings.setSettingsRepository(this);
138                 settingsMap.put(key, settings);
139                 updateChildren((ProvisionNode) o, settings);
140                 updateParents(settings);
141                 return settings;
142             } else {
143                 // It should be impossible to arrive here
144                 return null; // but let's keep the compiler happy.
145             }
146         }
147     }

149     /**
150      * This method loops through all the *potential* parents of 'settings'. There 3 possible cases:
151      * 
152      * 1) No entry for the potential parent of 'settings' exists
153      * 
154      * We create a ProvisionNode for this potential parent and insert 'settings' in that provision
155      * node.
156      * 
157      * 2) There entry is of type HierarchicalSettings for the potential parent.
158      * 
159      * The entry is settings nearest existing parent. We update settings parent field with this
160      * entry. We also break from the loop because updating our parent's parent is our parent's
161      * responsibility.
162      * 
163      * 3) There entry is of type ProvisionNode for this potential parent.
164      * 
165      * We add 'settings' to the list of children for this potential parent.
166      */
167     final private void updateParents(HierarchicalSettings settings) {
168         String name = settings.getName();
169         int length = name.length();
170         boolean parentFound = false;
171 
172         // System.out.println("UpdateParents called for " + name);
173 
174         // if name = "w.x.y.z", loop thourgh "w.x.y", "w.x" and "w", but not "w.x.y.z"
175         for (int i = name.lastIndexOf('.', length - 1); i >= 0; i = name.lastIndexOf('.', i - 1)) {
176             String substr = name.substring(0, i);
177 
178             LOG.debug("Updating parent : " + substr);
179             SettingsKey key = new SettingsKey(substr); // simple constructor
180             Object o = settingsMap.get(key);
181             // Create a provision node for a future parent.
182             if (o == null) {
183                 // System.out.println("No parent "+substr+" found. Creating ProvisionNode.");
184                 ProvisionNode pn = new ProvisionNode(settings);
185                 settingsMap.put(key, pn);
186             } else if (o instanceof HierarchicalSettings) {
187                 parentFound = true;
188                 settings.setParent((HierarchicalSettings) o);
189                 break; // no need to update the ancestors of the closest ancestor
190             } else if (o instanceof ProvisionNode) {
191                 ((ProvisionNode) o).addElement(settings);
192             } else {
193                 Exception e = new IllegalStateException("unexpected object type " + o.getClass() + " in ht.");
194                 e.printStackTrace();
195             }
196         }
197         // If we could not find any existing parents, then link with root.
198         if (!parentFound)
199             settings.setParent(root);
200     }

202     /**
203      * We update the links for all the children that placed themselves in the provision node 'pn'.
204      * The second argument 'settings' is a reference for the newly created HierarchicalSettings,
205      * parent of all the children in 'pn'
206      * 
207      * We loop on all the children 's' in 'pn':
208      * 
209      * If the child 's' has been already linked to a child of 'setting' then there is no need to
210      * update 's'.
211      * 
212      * Otherwise, we set settings parent field to s's parent and set s's parent field to settings.
213      * 
214      */
215     final private void updateChildren(ProvisionNode pn, HierarchicalSettings settings) {
216         // System.out.println("updateChildren called for " + settings.name);
217         final int last = pn.size();
218 
219         for (int i = 0; i < last; i++) {
220             HierarchicalSettings s = (HierarchicalSettings) pn.elementAt(i);
221             // System.out.println("Updating child " +s.name);
222 
223             // Unless this child already points to a correct (lower) parent,
224             // make settings.parent point to s.parent and s.parent to settings.
225             if (!s.getParent().getName().startsWith(settings.getName())) {
226                 settings.setParent(s.getParent());
227                 s.setParent(settings);
228             }
229         }
230     }

240     /** {@inheritDoc} */
241     public void resetConfiguration() {
242 
243         root.setAdditivity(true);
244         root.setParent(null);
245         root.setMapping(null);
246         root.setSettingsRepository(null);
247         root.removeAllConnectors();
248 
249         Iterator it = settingsMap.entrySet().iterator();
250         while (it.hasNext()) {
251             HierarchicalSettings s = (HierarchicalSettings) it.next();
252             s.setAdditivity(true);
253             s.setParent(null);
254             s.setMapping(null);
255             s.setSettingsRepository(null);
256             s.removeAllConnectors();
257         }
258         settingsMap.clear();
259         
260         it = connectorMap.entrySet().iterator();
261         while (it.hasNext()) {
262             Connector c = (Connector) it.next();
263             c.setContentResolver(null);
264             c.setObjectResolver(null);
265         }
266         connectorMap.clear();
267     }
