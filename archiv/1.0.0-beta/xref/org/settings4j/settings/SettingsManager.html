17  package org.settings4j.settings;
18  
19  import java.net.URL;
20  import java.util.List;
21  
22  import org.settings4j.Settings;
23  import org.settings4j.SettingsFactory;
24  import org.settings4j.SettingsRepository;
25  import org.settings4j.config.DOMConfigurator;
26  import org.settings4j.contentresolver.ClasspathContentResolver;
27  import org.settings4j.settings.nop.NOPSettingsRepository;
28  
29  /**
30   * managed The {@link SettingsRepository} .
31   * This {@link SettingsRepository} is used to store the configuration from the settings4j.xml.
32   * 
33   * @author hbrabenetz
34   *
35   */
36  public class SettingsManager {
37      /** General Logger for this Class */
38      private static final org.apache.commons.logging.Log LOG = org.apache.commons.logging.LogFactory
39          .getLog(SettingsManager.class);
40  
41      /**
42       * The Classpath-Location of the settings4j.xml who i readed by default <br />
43       * settings4j.xml
44       */
45      public static final String DEFAULT_XML_CONFIGURATION_FILE = "settings4j.xml";
46  
47      /**
48       * The fallback settings4j.xml who is used if not settings4j.xml where found. <br />
49       * org/settings4j/config/defaultsettings4j.xml
50       */
51      public static final String DEFAULT_FALLBACK_CONFIGURATION_FILE = "org/settings4j/config/defaultsettings4j.xml";
52  
53      /**
54       * The internal default Settings4j Repository where all {@link org.settings4j.Settings} are stored.
55       */
56      private static SettingsRepository settingsRepository;
57      static {
58          /**
59           * The root settings names itself as "root". However, the root settings cannot be retrieved
60           * by name.
61           */
62          settingsRepository = new HierarchicalSettingsRepository(new DefaultSettings("root"));
63  
64          // read XML default Configuration to configure the repository
65          URL url = ClasspathContentResolver.getResource(DEFAULT_XML_CONFIGURATION_FILE);
66  
67          // If we have a non-null url, then delegate the rest of the
68          // configuration to the DOMConfigurator.configure method.
69          if (url != null) {
70              LOG.debug("Using URL [" + url + "] for automatic settings4j configuration.");
71              try {
72                  DOMConfigurator.configure(url, settingsRepository);
73              } catch (NoClassDefFoundError e) {
74                  LOG.warn("Error during default initialization", e);
75              }
76          } else {
77              LOG.debug("Could not find resource: [" + DEFAULT_XML_CONFIGURATION_FILE + "].");
78          }
79      }
80  
81      /**
82       * The internal default Settings4j Repository where all {@link org.settings4j.Settings} are stored.
83       * 
84       * @return The Settings4j Repository, Returns NEVER null put maybe a {@link NOPSettingsRepository}.
85       */
86      public static SettingsRepository getSettingsRepository() {
87          if (settingsRepository == null) {
88              settingsRepository = new NOPSettingsRepository();
89              LOG.error("SettingsManager.settingsRepository was null likely due to error in class reloading, using NOPSettingsRepository.");
90          }
91          return settingsRepository;
92      }
93  
94      /**
95       * the root {@link org.settings4j.Settings} <br />
96       * e.g.: Defined by the root-Tag inside the settings4j.xml
97       * 
98       * @return
99       */
100     public static Settings getRootSettings() {
101         return getSettingsRepository().getRootSettings();
102     }
103 
104     /**
105      * Retrieve the appropriate {@link Settings} instance.
106      */
107     public static Settings getSettings(final String name) {
108         // Delegate the actual manufacturing of the settings to the settings repository.
109         return getSettingsRepository().getSettings(name);
110     }
111 
112     /**
113      * Retrieve the appropriate {@link Settings} instance.
114      */
115     public static Settings getSettings(final Class clazz) {
116         // Delegate the actual manufacturing of the settings to the settings repository.
117         return getSettingsRepository().getSettings(clazz.getName());
118     }
119 
120     /**
121      * Retrieve the appropriate {@link Settings} instance.
122      */
123     public static Settings getSettings(final String name, final SettingsFactory factory) {
124         // Delegate the actual manufacturing of the settings to the settings repository.
125         return getSettingsRepository().getSettings(name, factory);
126     }
127 
128     /**
129      * Checks, if the Settings-Object for the given key already exists.<br />
130      * The difference to {@link #getSettings(Class)}: no Settings-Object will be created if it doesn't exists.
131      * 
132      * @param name The name of The {@link Settings}
133      * @return the founded Settings Object or null.
134      */
135     public static Settings exists(final String name) {
136         return getSettingsRepository().exists(name);
137     }
138 
139     /**
140      * return a List off all {@link org.settings4j.Settings} who are defind in this Repository.
141      * 
142      * @return the List of all defined Settings, returns NEVER null but maybe an empty list.
143      */
144     public static List getCurrentSettingsList() {
145         return getSettingsRepository().getCurrentSettingsList();
146     }
147 }
