141     /**
142      * Sets a parameter based from configuration file content.
143      * 
144      * @param elem param element, may not be null.
145      * @param propSetter property setter, may not be null.
146      * @param props properties
147      */
148     private void setParameter(final Element elem, final Object bean, Connector[] connectors) {
149         String name = elem.getAttribute("name");
150         String valueStr = (elem.getAttribute("value"));
151         try {
152             PropertyDescriptor propertyDescriptor = PropertyUtils.getPropertyDescriptor(bean, name);
153             Method setter = PropertyUtils.getWriteMethod(propertyDescriptor);
154             Object value;
155             if(connectors != null){
156                 value = subst(valueStr, connectors, setter.getParameterTypes()[0]);
157             } else {
158                 value = subst(valueStr, null, setter.getParameterTypes()[0]);
159             }
160             PropertyUtils.setProperty(bean, name, value);
161         } catch (IllegalAccessException e) {
162             LOG.warn("Cannnot set Property: " + name, e);
163         } catch (InvocationTargetException e) {
164             LOG.warn("Cannnot set Property: " + name, e);
165         } catch (NoSuchMethodException e) {
166             LOG.warn("Cannnot set Property: " + name, e);
167         }
168     }

177     public void doConfigure(final URL url) {
178         ParseAction action = new ParseAction() {
179             public Document parse(final DocumentBuilder parser) throws SAXException, IOException {
180                 return parser.parse(url.toString());
181             }
182 
183             public String toString() {
184                 return "url [" + url.toString() + "]";
185             }
186         };
187         doConfigure(action);
188     }

190     private final void doConfigure(final ParseAction action) throws FactoryConfigurationError {
191         DocumentBuilderFactory dbf = null;
192         try {
193             LOG.debug("System property is :" + new SystemPropertyConnector().getString(dbfKey));
194             dbf = DocumentBuilderFactory.newInstance();
195             LOG.debug("Standard DocumentBuilderFactory search succeded.");
196             LOG.debug("DocumentBuilderFactory is: " + dbf.getClass().getName());
197         } catch (FactoryConfigurationError fce) {
198             Exception e = fce.getException();
199             LOG.debug("Could not instantiate a DocumentBuilderFactory.", e);
200             throw fce;
201         }
202 
203         try {
204             dbf.setValidating(true);
205 
206             DocumentBuilder docBuilder = dbf.newDocumentBuilder();
207 
208             docBuilder.setErrorHandler(new SAXErrorHandler());
209             docBuilder.setEntityResolver(new Settings4jEntityResolver());
210 
211             Document doc = action.parse(docBuilder);
212             parse(doc.getDocumentElement());
213         } catch (Exception e) {
214             // I know this is miserable...
215             LOG.error("Could not parse " + action.toString() + ".", e);
216         }
217     }

220      * Used internally to configure the settings4j framework by parsing a DOM tree of XML elements based
221      * on <a href="doc-files/settings4j.dtd">settings4j.dtd</a>.
222      * 
223      */
224     protected void parse(Element element) {
225 
226         String rootElementName = element.getTagName();
227 
228         if (!rootElementName.equals(CONFIGURATION_TAG)) {
229             LOG.error("DOM element is - not a <" + CONFIGURATION_TAG + "> element.");
230             return;
231         }
232 
233         // TODO hbrabenetz 15.06.2008 : reset Configuration
234 //        //
235 //        //   reset repository before configuration if reset="true"
236 //        //       on configuration element.
237 //        //
238 //      String resetAttrib = subst(element.getAttribute(RESET_ATTR));
239 //      LogLog.debug("reset attribute= \"" + resetAttrib +"\".");
240 //      if(!("".equals(resetAttrib))) {
241 //           if (OptionConverter.toBoolean(resetAttrib, false)) {
242 //               repository.resetConfiguration();
243 //           }
244 //      }
245         
246         
247         /*
248          * Building Appender objects, placing them in a local namespace for future reference
249          */
250 
251         // First configure each category factory under the root element.
252         // Category factories need to be configured before any of
253         // categories they support.
254         //
255         String tagName = null;
256         Element currentElement = null;
257         Node currentNode = null;
258         NodeList children = element.getChildNodes();
259         final int length = children.getLength();
260 
261         for (int loop = 0; loop < length; loop++) {
262             currentNode = children.item(loop);
263             if (currentNode.getNodeType() == Node.ELEMENT_NODE) {
264                 currentElement = (Element) currentNode;
265                 tagName = currentElement.getTagName();
266 
267                 if (tagName.equals(SETTINGS_TAG)) {
268                     parseSettings(currentElement);
269                 } else if (tagName.equals(ROOT_TAG)) {
270                     parseRoot(currentElement);
271                 }
272             }
273         }
274     }

276     /**
277      * Used internally to parse an category element.
278      */
279     protected void parseSettings(Element loggerElement) {
280         // Create a new org.settings4j.Settings object from the <settings> element.
281         String settingsName = loggerElement.getAttribute(NAME_ATTR);
282 
283         Settings settings;
284 
285         LOG.debug("Retreiving an instance of org.settings4j.Settings");
286         // settings = (catFactory == null) ? repository.getSettings(settingsName) :
287         // repository.getSettings(settingsName, catFactory);
288         settings = repository.getSettings(settingsName);
289 
290         // Setting up a category needs to be an atomic operation, in order
291         // to protect potential log operations while category
292         // configuration is in progress.
293         synchronized (settings) {
294             Boolean additivity = (Boolean)subst(loggerElement.getAttribute(ADDITIVITY_ATTR), null, Boolean.class);
295 
296             LOG.debug("Setting [" + settings.getName() + "] additivity to [" + additivity + "].");
297             if (additivity != null ){
298                 settings.setAdditivity(additivity.booleanValue());
299             }
300             parseChildrenOfSettingsElement(loggerElement, settings, false);
301         }
302     }

304     /**
305      * Used internally to parse an category element.
306      */
307     protected Connector parseConnector(Element connectorElement) {
308         String connectorName = connectorElement.getAttribute(NAME_ATTR);
309 
310         Connector connector;
311 
312         String className = connectorElement.getAttribute(CLASS_ATTR);
313 
314         
315         LOG.debug("Desired Connector class: [" + className + ']');
316         try {
317             Class clazz = loadClass(className);
318             Constructor constructor = clazz.getConstructor(NO_PARAM);
319             connector = (Connector) constructor.newInstance(null);
320         } catch (Exception oops) {
321             LOG.error("Could not retrieve connector [" + connectorName + "]. Reported error follows.", oops);
322             return null;
323         }
324         
325         connector.setName(connectorName);
326         
327         Connector[] subConnectors = getConnectors(connectorElement);
328         for (int i = 0; i < subConnectors.length; i++) {
329             connector.addConnector(subConnectors[i]);
330         }
331         
332         // Setting up a category needs to be an atomic operation, in order
333         // to protect potential log operations while category
334         // configuration is in progress.
335         synchronized (connector) {
336 
337             NodeList children = connectorElement.getChildNodes();
338             final int length = children.getLength();
339 
340             for (int loop = 0; loop < length; loop++) {
341                 Node currentNode = children.item(loop);
342 
343                 if (currentNode.getNodeType() == Node.ELEMENT_NODE) {
344                     Element currentElement = (Element) currentNode;
345                     String tagName = currentElement.getTagName();
346 
347                     if (tagName.equals(CONTENT_RESOLVER_REF_TAG)) {
348                         Element contentResolverRef = (Element) currentNode;
349                         ContentResolver contentResolver = findContentResolverByReference(contentResolverRef);
350                         connector.setContentResolver(contentResolver);
351 
352                     } else if (tagName.equals(OBJECT_RESOLVER_REF_TAG)) {
353                         Element objectResolverRef = (Element) currentNode;
354                         ObjectResolver objectResolver = findObjectResolverByReference(objectResolverRef);
355                         connector.setObjectResolver(objectResolver);
356 
357                     } else if (tagName.equals(PARAM_TAG)) {
358                         setParameter(currentElement, connector, subConnectors);
359                     } else {
360                         quietParseUnrecognizedElement(connector, currentElement);
361                     }
362                 }
363             }
364             
365             if (connector instanceof AbstractConnector){
366                 connector = new ContentHasChangedNotifierConnectorWrapper((AbstractConnector)connector);
367             }
368 
369             Boolean cached = (Boolean)subst(connectorElement.getAttribute(CACHED_ATTR), null, Boolean.class);
370             if (cached != null && cached.booleanValue()){
371                 connector = new CachedConnectorWrapper(connector);
372             }
373             
374             // initial the connector
375             connector.init();
376         }
377         return connector;
378     }

380     /**
381      * Only logs out unrecognized Elements
382      * 
383      * @param instance instance, may be null.
384      * @param element element, may not be null.
385      */
386     private static void quietParseUnrecognizedElement(final Object instance, final Element element) {
387         String elementName = "UNKNOWN";
388         String instanceClassName = "UNKNOWN";
389         
390         try{
391             elementName = element.getNodeName();
392             instanceClassName = instance.getClass().getName();
393         }catch (Exception e) {
394             LOG.warn("Error in quietParseUnrecognizedElement(): " + e.getMessage());
395             if (LOG.isDebugEnabled()){
396                 LOG.debug(e.getMessage(), e);
397             }
398         }
399         LOG.warn("Unrecognized Element will be ignored: " + elementName + " for Instance: " + instanceClassName);
400     }

402     /**
403      * Return all referenced connectors from Child-Nodes {@link #CONNECTOR_REF_TAG}
404      * 
405      * @param element
406      * @return
407      */
408     protected Connector[] getConnectors(Element element){
409         List connectors = new ArrayList();
410         
411         NodeList children = element.getChildNodes();
412         final int length = children.getLength();
413 
414         for (int loop = 0; loop < length; loop++) {
415             Node currentNode = children.item(loop);
416 
417             if (currentNode.getNodeType() == Node.ELEMENT_NODE) {
418                 Element currentElement = (Element) currentNode;
419                 String tagName = currentElement.getTagName();
420 
421                 if (tagName.equals(CONNECTOR_REF_TAG)) {
422                     Element connectorRef = (Element) currentNode;
423                     Connector connector = findConnectorByReference(connectorRef);
424                     connectors.add(connector);
425                 }
426             }
427         }
428         
429         return (Connector[])connectors.toArray(new Connector[connectors.size()]);
430     }

444     /**
445      * Used internally to parse the children of a category element.
446      */
447     protected void parseChildrenOfSettingsElement(Element settingsElement, Settings settings, boolean isRoot) {
448         String mappingRef = settingsElement.getAttribute(MAPPING_REF_ATTR);
449         if (!StringUtils.isEmpty(mappingRef)){
450             Map mapping = findMappingByName(settingsElement.getOwnerDocument(), mappingRef);
451             if (mapping != null){
452                 settings.setMapping(mapping);
453             }
454         }
455         
456         // Remove all existing appenders from settings. They will be
457         // reconstructed if need be.
458         settings.removeAllConnectors();
459 
460         NodeList children = settingsElement.getChildNodes();
461         final int length = children.getLength();
462 
463 
464         Connector[] connectors = getConnectors(settingsElement);
465         for (int i = 0; i < connectors.length; i++) {
466             settings.addConnector(connectors[i]);
467         }
468         
469         for (int loop = 0; loop < length; loop++) {
470             Node currentNode = children.item(loop);
471 
472             if (currentNode.getNodeType() == Node.ELEMENT_NODE) {
473                 Element currentElement = (Element) currentNode;
474                 String tagName = currentElement.getTagName();
475 
476                 if (tagName.equals(PARAM_TAG)) {
477                     setParameter(currentElement, settings, connectors);
478                 } else if (tagName.equals(CONNECTOR_REF_TAG)) {
479                     LOG.debug("Connector already added");
480                 } else {
481                     quietParseUnrecognizedElement(settings, currentElement);
482                 }
483             }
484         }
485     }

489     /**
490      * Used internally to parse connectors by IDREF name.
491      */
492     protected Connector findConnectorByName(Document doc, String connectorName) {
493         Connector connector = (Connector) connectorBag.get(connectorName);
494 
495         if (connector != null) {
496             return connector;
497         } else {
498             Element element = getElementByNameAttr(doc, connectorName, "connector");
499 
500             if (element == null) {
501                 LOG.error("No connector named [" + connectorName + "] could be found.");
502                 return null;
503             } else {
504                 connector = parseConnector(element);
505                 connectorBag.put(connectorName, connector);
506                 return connector;
507             }
508         }
509     }

511     /**
512      * Used internally to parse connectors by IDREF element.
513      */
514     protected Connector findConnectorByReference(Element connectorRef) {
515         String connectorName = connectorRef.getAttribute(REF_ATTR);
516         Boolean readonly = (Boolean)subst(connectorRef.getAttribute(READONLY_ATTR), null, Boolean.class);
517         Document doc = connectorRef.getOwnerDocument();
518         Connector connector = findConnectorByName(doc, connectorName);
519         if (readonly != null && readonly.booleanValue()){
520             return new ReadOnlyConnectorWrapper(connector);
521         }
522         return connector;
523     }

534         String className = objectResolverElement.getAttribute(CLASS_ATTR);

554 //        }

557         // to protect potential settings operations while settings
558         // configuration is in progress.
559         synchronized (objectResolver) {
560 
561             NodeList children = objectResolverElement.getChildNodes();
562             final int length = children.getLength();
563 
564             for (int loop = 0; loop < length; loop++) {
565                 Node currentNode = children.item(loop);
566 
567                 if (currentNode.getNodeType() == Node.ELEMENT_NODE) {
568                     Element currentElement = (Element) currentNode;
569                     String tagName = currentElement.getTagName();
570 
571                     if (tagName.equals(OBJECT_RESOLVER_REF_TAG)) {
572                         Element objectResolverRef = (Element) currentNode;
573                         ObjectResolver subObjectResolver = findObjectResolverByReference(objectResolverRef);
574                         objectResolver.addObjectResolver(subObjectResolver);
575 
576                     } else if (tagName.equals(PARAM_TAG)) {
577                         setParameter(currentElement, objectResolver, connectors);
578                     } else {
579                         quietParseUnrecognizedElement(objectResolver, currentElement);
580                     }
581                 }
582             }
583 
584             Boolean cached = (Boolean)subst(objectResolverElement.getAttribute(CACHED_ATTR), null, Boolean.class);
585             if (cached != null && cached.booleanValue()){
586                 if (objectResolver instanceof AbstractObjectResolver){
587                     ((AbstractObjectResolver)objectResolver).setCached(cached.booleanValue());
588                 } else {
589                     LOG.warn("Only AbstractObjectResolver can use the attribute cached=\"true\" ");
590                     // TODO hbrabenetz 21.05.2008 : extract setCached into seperate Interface.
591                 }
592             }
593         }
594         
595         return objectResolver;
596     }

622             return null;
623         }
624         
625         // Setting up a mapping needs to be an atomic operation, in order
626         // to protect potential settings operations while settings
627         // configuration is in progress.
628         synchronized (mapping) {
629 
630             NodeList children = mappingElement.getChildNodes();
631             final int length = children.getLength();
632 
633             for (int loop = 0; loop < length; loop++) {
634                 Node currentNode = children.item(loop);
635 
636                 if (currentNode.getNodeType() == Node.ELEMENT_NODE) {
637                     Element currentElement = (Element) currentNode;
638                     String tagName = currentElement.getTagName();
639 
640                     if (tagName.equals(ENTRY_TAG)) {
641                         String key = currentElement.getAttribute(ENTRY_KEY_ATTR);

643                         mapping.put(key, keyRef);
644                     } else {
645                         quietParseUnrecognizedElement(mapping, currentElement);
646                     }
647                 }
648             }
649         }
650         
651         return mapping;
652     }

819         // Doesn't work on DOM Level 1 :

821 
822         // Endre's hack:
823         Element element = null;
824         NodeList list = doc.getElementsByTagName(elementTagName);
825         for (int t = 0; t < list.getLength(); t++) {
826             Node node = list.item(t);
827             NamedNodeMap map = node.getAttributes();
828             Node attrNode = map.getNamedItem("name");
829             if (nameAttrValue.equals(attrNode.getNodeValue())) {

857     /**
858      * In future implementation of this function will replace expressions like ${connectors.object['']}
859      * or simply ${true}
860      * 
861      * @param value
862      * @param clazz
863      * @param connectors
864      * @return
865      */
866     protected Object subst(final String value, Connector[] connectors, final Class clazz) {
867         if (StringUtils.isEmpty(value)){
868             return null;
869         }
870         
871         if (value.contains("${")){
872             try {
873                 Map context = new HashMap(expressionAttributes);
874                 if (connectors != null){
875                     // Expression like ${connectors.object['...']} or ${connectors.string['...']} 
876                     context.put("connectors", new ELConnectorWrapper(connectors));
877 
878                     // Expression like ${connector.FsConnector.object['...']} or ${connector.ClassPathConnector.string['...']} 
879                     Map connectorMap = new HashMap();
880                     for (int i = 0; i < connectors.length; i++) {
881                         Connector connector = connectors[i];
882                         connectorMap.put(connector.getName(), new ELConnectorWrapper(new Connector[]{connector}));
883                     }
884                     context.put("connector", connectorMap);
885                 }
886                 // Expression like ${env['...']} e.g.:  ${env['TOMCAT_HOME']} or ${env.TOMCAT_HOME}
887                 context.put("env", System.getenv());
888                 Object result = ExpressionLanguageUtil.evaluateExpressionLanguage(value, context, clazz);
889                 return result;
890             } catch (ELException e) {
891                 LOG.error(e.getMessage(), e);
892                 return null;
893             }
894         } else {
895             if (clazz.equals(String.class)){
896                 return value.trim();
897             } else if (clazz.equals(Boolean.class)){
898                 String trimmedVal = value.trim();
899                 if("true".equalsIgnoreCase(trimmedVal)){
900                   return Boolean.TRUE;
901                 } else {
902                   //if("false".equalsIgnoreCase(trimmedVal))
903                   return Boolean.FALSE;
904                 }
905             } else {
906                 throw new UnsupportedOperationException("The following Type is not supported now: " + clazz + "; found value: " + value);
907             }
908         }
909     }
