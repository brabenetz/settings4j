17  package org.settings4j;

19  import org.settings4j.objectresolver.JavaXMLBeansObjectResolver;
20  import org.settings4j.objectresolver.ReadOnlyObjectResolverWrapper;
21  import org.settings4j.objectresolver.UnionObjectResolver;

23  /**
24   * The ObjectResolver is a Helper to resolce byte[] Content to a Java Object.<br />
25   * <br />
26   * <pre>
27   * Example:
28   * <div style="border-width:1px;border-style:solid;">
29   * &lt;connector name="<span style="color:green;">ClasspathConnector</span>" class="org.settings4j.connector.ClasspathConnector"&gt;
30   *     &lt;objectResolver-ref ref="<span style="color:red;">JavaXMLBeansObjectResolver</span>" /&gt;
31   * &lt;/connector&gt;
32   * <span style="color:blue;">&lt;root&gt;</span>
33   *     &lt;connector-ref ref="<span style="color:green;">ClasspathConnector</span>" /&gt;
34   * <span style="color:blue;">&lt;/root&gt;</span>
35   * &lt;objectResolver name="<span style="color:red;">JavaXMLBeansObjectResolver</span>" class="org.settings4j.objectresolver.JavaXMLBeansObjectResolver" /&gt;
36   * </div></pre>
37   * 
38   * 
39   * You can now store a XMLEncoded (Serialized) Java-Objects into your Classpath. see {@link JavaXMLBeansObjectResolver} for more details.<br />
40   * <br />
41   * Most ObjectResolver-implementations needs a property-File for more informations, how the XML-File should be resolved.<br />
42   * By convention the propertyfile will be searched by the given contentResolver under key + ".properties"
43   * 
44   * 
45   * @author hbrabenetz
46   *
47   */
48  public interface ObjectResolver {
49      
50      /**
51       * Reads the byte[] content from the ContentResolver and creates an Object.
52       * 
53       * The normal usecase of an implementation of this ObjectResolver Interface:<br />
54       * 
55       * <ol>
56       * <li>Read the Propertyfile from {@link ContentResolver}.getContent(key + ".properties")</li>
57       * <li>Read the Value of "objectResolverKey" from propertyfile</li>
58       * <li>The "objectResolverKey" defines which ObjectResolver-Implementation should solve the byte[] content to a Java-Object</li>
59       * <li>Convert the byte[] from {@link ContentResolver#getContent(String key)} to an Object.</li>
60       * <li>Maybe additional values are consumed from the propertyfile</li>
61       * </ol>
62       * 
63       * 
64       * @param key The Key of the byte[] who should be converted to an Object.
65       * @param contentResolver The contentResolver, from where the content could be read.
66       * @return the Object, or null if this Object-Resolver can not convert the byte[] Content to an Object.
67       */
68      public Object getObject(String key, ContentResolver contentResolver);
69      
70      /**
71       * Serialize and store the byte[] with the ContentResolver under the given key.
72       * 
73       * The normal usecase of an implementation of this ObjectResolver Interface:<br />
74       * 
75       * <ol>
76       * <li>Read the Propertyfile from {@link ContentResolver}.getContent(key + ".properties")</li>
77       * <li>Read the Value of "objectResolverKey" from propertyfile</li>
78       * <li>The "objectResolverKey" defines which ObjectResolver-Implementation should store the Java-Object to an Byte[]</li>
79       * <li>Convert the Object to a byte[] and store it with {@link ContentResolver#setContent(String, byte[])} .</li>
80       * <li>Maybe additional values are consumed from the propertyfile</li>
81       * </ol>
82       * 
83       * @param key The Key of the byte[] the serialized Object should be stored.
84       * @param contentResolver The ContentResolver is used to store the byte[] content.
85       * @param value The new Object who should be serialized.
86       * @return {@link Constants#SETTING_SUCCESS} or {@link Constants#SETTING_NOT_POSSIBLE}
87       */
88      public int setObject(String key, ContentResolver contentResolver, Object value);
89  
90      /**
91       * Some Implementations of a {@link ObjectResolver} are delegating the functionality
92       * to other ObjectResolvers.<br />
93       * Examples are: {@link ReadOnlyObjectResolverWrapper} or {@link UnionObjectResolver}
94       * 
95       * <pre>
96       * --------------------------------------
97       * &lt;objectResolver name="DefaultObjectResolver" class="org.settings4j.objectresolver.UnionObjectResolver"&gt;
98       *     &lt;objectResolver-ref ref="JavaXMLBeansObjectResolver" /&gt;
99       *     &lt;objectResolver-ref ref="SpringConfigObjectResolver" /&gt;
100      * &lt;/objectResolver&gt;
101      * --------------------------------------
102      * </pre>
103      * 
104      * @param objectResolver the original objectResolver to delegate.
105      */
106     public void addObjectResolver(ObjectResolver objectResolver);
107 
108     
109     /**
110      * Notify the ObjectResolver-Implementation if a byte[] content is stored by some ContentResolver<br />
111      * <br />
112      * This is the right place to clear cached Objects for the given key.<br />
113      * 
114      * @param key The key from which the Content has changed.
115      */
116     public void notifyContentHasChanged(String key);
117 }
