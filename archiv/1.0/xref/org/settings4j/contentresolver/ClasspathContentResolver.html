17  package org.settings4j.contentresolver;
18  
19  import java.io.IOException;
20  import java.io.InputStream;
21  import java.net.URL;
22  
23  import org.apache.commons.io.IOUtils;
24  import org.settings4j.ContentResolver;
25  
26  /**
27   * {@link ContentResolver} implementation to read content from the Classpath.
28   * <p>
29   * Uses the default ClassLoader: typically the thread context ClassLoader
30   * see {@link #getClassLoader()}.
31   * <p>
32   * Optional Path Prefix is "classpath:".
33   * 
34   * @author Harald.Brabenetz
35   *
36   */
37  public class ClasspathContentResolver implements ContentResolver {
38  
39      /** General Logger for this Class. */
40      private static final org.apache.commons.logging.Log LOG = org.apache.commons.logging.LogFactory
41          .getLog(ClasspathContentResolver.class);
42  
43  
44      /** Pseudo URL prefix for loading from the class path: "classpath:". */
45      public static final String CLASSPATH_URL_PREFIX = "classpath:";
46  
47  
48      /** {@inheritDoc} */
49      public void addContentResolver(final ContentResolver contentResolver) {
50          throw new UnsupportedOperationException("ClasspathContentResolver cannot add other ContentResolvers");
51      }
52  
53      /** {@inheritDoc} */
54      public byte[] getContent(final String key) {
55          final String normalizedKey = normalizeKey(key);
56  
57          try {
58              final InputStream is = getClassLoader().getResourceAsStream(normalizedKey);
59              if (is != null) {
60                  return IOUtils.toByteArray(is);
61              }
62              // else
63              return null;
64          } catch (final IOException e) {
65              LOG.error(e.getMessage(), e);
66              return null;
67          }
68      }
69  
70      /**
71       * Method to get onlx the URL for the given Key.
72       * <p>
73       * 
74       * @param key the key (could have a 'classpath:' prefix or not)
75       * @return The {@link URL}, see {@link ClassLoader#getResource(String)}.
76       */
77      public static URL getResource(final String key) {
78          final String normalizedKey = normalizeKey(key);
79  
80          return getClassLoader().getResource(normalizedKey);
81      }
82  
83  
84      /**
85       * Return the default ClassLoader to use: typically the thread context ClassLoader, if available; the ClassLoader
86       * that loaded the ClasspathContentResolver class will be used as fallback.
87       * <p>
88       * Call this method if you intend to use the thread context ClassLoader in a scenario where you absolutely need a
89       * non-null ClassLoader reference: for example, for class path resource loading (but not necessarily for
90       * <code>Class.forName</code>, which accepts a <code>null</code> ClassLoader reference as well).
91       * 
92       * @return the default ClassLoader (never <code>null</code>)
93       * @see java.lang.Thread#getContextClassLoader()
94       */
95      public static ClassLoader getClassLoader() {
96          ClassLoader cl = null;
97          try {
98              cl = Thread.currentThread().getContextClassLoader();
99          } catch (final Throwable ex) {
100             LOG.debug("Cannot access thread context ClassLoader - falling back to system class loader", ex);
101         }
102         if (cl == null) {
103             // No thread context class loader -> use class loader of this class.
104             cl = ClasspathContentResolver.class.getClassLoader(); // NOPMD it is only a fallback case.
105         }
106         return cl;
107     }
108 
109     private static String normalizeKey(final String key) {
110         String normalizedKey = key;
111         if (normalizedKey.startsWith(CLASSPATH_URL_PREFIX)) {
112             normalizedKey = normalizedKey.substring(CLASSPATH_URL_PREFIX.length());
113         }
114         if (normalizedKey.startsWith("/")) {
115             normalizedKey = normalizedKey.substring(1);
116         }
117         return normalizedKey;
118     }
119 }
