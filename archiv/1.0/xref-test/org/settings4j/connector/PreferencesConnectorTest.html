17  package org.settings4j.connector;
18  
19  import java.util.prefs.BackingStoreException;
20  import java.util.prefs.Preferences;
21  
22  import org.settings4j.Settings4j;
23  
24  import junit.framework.TestCase;
25  
26  /**
27   * Test Suite for {@link PreferencesConnector}.
28   * Checkstyle:OFF .*
29   * 
30   * @author brabenetz
31   *
32   */
33  public class PreferencesConnectorTest extends TestCase {
34  
35  
36      private static final String PREF_UNITTEST_NODE = "org/settings4j/unittest";
37  
38      protected void setUp() throws Exception {
39          removeUnitTestNode(Preferences.userRoot());
40          removeUnitTestNode(Preferences.systemRoot());
41      }
42  
43      private void removeUnitTestNode(final Preferences userRoot) throws BackingStoreException {
44          if (userRoot.nodeExists(PREF_UNITTEST_NODE)) {
45              userRoot.node(PREF_UNITTEST_NODE).removeNode();
46          }
47      }
48  
49      public void testSaveSuccessSlash() {
50          final PreferencesConnector connector = new PreferencesConnector();
51          connector.setString(PREF_UNITTEST_NODE + "/testSaveSuccessSlash", "test");
52  
53          assertEquals("test", Preferences.userRoot().node(PREF_UNITTEST_NODE).get("testSaveSuccessSlash", null));
54      }
55      
56      public void testSaveSuccessBackslash() {
57          final PreferencesConnector connector = new PreferencesConnector();
58          connector.setString(PREF_UNITTEST_NODE + "\\testSaveSuccessBackslash", "test");
59  
60          assertEquals("test", Preferences.userRoot().node(PREF_UNITTEST_NODE).get("testSaveSuccessBackslash", null));
61      }
62  
63      public void testReadSuccess() {
64          Preferences.userRoot().node(PREF_UNITTEST_NODE).put("testReadSuccess", "test");
65          
66          final PreferencesConnector connector = new PreferencesConnector();
67  
68          assertEquals("test", connector.getString(PREF_UNITTEST_NODE + "/testReadSuccess"));
69      }
70  
71      public void testReadFail() {
72          final PreferencesConnector connector = new PreferencesConnector();
73  
74          assertNull(connector.getString(PREF_UNITTEST_NODE + "/testReadFailDoesntExist"));
75          assertNull(connector.getString("testReadFailDoesntExist"));
76      }
77      
78      public void testRoundup() {
79          final PreferencesConnector connector = new PreferencesConnector();
80          
81          assertNull(connector.getString(PREF_UNITTEST_NODE + "/testRoundup"));
82          // 1. save
83          connector.setString(PREF_UNITTEST_NODE + "/testRoundup", "test");
84          // 2. read
85          assertEquals("test", connector.getString(PREF_UNITTEST_NODE + "/testRoundup"));
86      }
87  
88      public void testFormularExample() {
89          final PreferencesConnector connector = new PreferencesConnector();
90          connector.setString("com/mycompany/mycalculation/my-formula", "a * a - 2");
91  
92          assertEquals("a * a - 2", Settings4j.getString("com/mycompany/mycalculation/my-formula"));
93      }
94  }
